import numpy as np
from scipy import ndimage as spnd

from CryoNIRSP_Pipeline.Align import Align3Calibration
from CryoNIRSP_Pipeline.Gain import Gain4Calibration
from CryoNIRSP_Pipeline.common import Data, generic
from CryoNIRSP_Pipeline.common.tag import tag

    [docs]def main(datasetData, pos, outpre, dark_cal=None, instrument_pol_cal=None, align3_cal=None, gain4_cal=None,
         gain4_shifts=None, ispol=True, telescope_db=None, remove_TM=True, include_M12=True, remove_parallactic=True):
    """Note that this script does NOT run on non-linearized data
    """

    print('{}: Loading pos {}'.format(tag(), pos))
    sciData = datasetData.get_slit_pos_data(pos)

    validate_observe_task_data(sciData)

    if dark_cal is not None:
        print('{}: doing dark correction'.format(tag()))
        generic.dark_correction(sciData, dark_cal)

    if gain4_cal is not None:
        print('{}: doing gain4 correction'.format(tag()))
        if ispol:
            generic.modulated_gain_correction(sciData, gain4_cal)
        else:
            modavg_gain = generic.average_FitsData(gain4_cal)
            generic.unmodulated_gain_correction(sciData, modavg_gain)

    if ispol and instrument_pol_cal is not None:
        print('{}: applying polarization calibration'.format(tag()))
        sciData = pol_corrections.polarization_correction(sciData, instrument_pol_cal, telescope_db=telescope_db,
                                                          remove_TM=remove_TM, include_M12=include_M12,
                                                          remove_parallactic=remove_parallactic)

    if align3_cal is not None:
        if gain4_shifts is None:
            raise ValueError('Cannot perform geometric correction without both Gain4 Shifts and Align3 Cals')

        if ispol:
            print('{}: Combining polarimetric beams'.format(tag()))
            sciData = combine_polarimetric_beams(sciData, align3_cal)
        else:
            print('{} Combining intensity beams'.format(tag()))
            sciData = combine_intensity_beams(sciData, align3_cal)

        print('{}: Removing spectral tilt and curvature'.format(tag()))
        sciData = geometric_correction(sciData, align3_cal, gain4_shifts)

    sci_cal = generate_science_calibration(sciData, outpre, pos)

    return sci_cal

    IP0 = fitsData.headers[1][S122['InstrumentProgramID']]
    for h in fitsData.data_hdus:
        if h.header[S122['InstrumentProgramID']] != IP0:

    [docs]def combine_polarimetric_beams(stokesData, align3_cal):
    """Transform Cryo beam 2 into beam 1 coordinates and then add the beams together.

    Each beam is first normalized by its Stokes I magnitude and the final, combined beam then has this normalization
    removed.

    Parameters
    ----------
    stokesData : `Data.FitsData`
        Object containing demodulated Cryo data

    align3_cal : dict
        Dictionary with Affine transform parameters that map beam 2 -> beam 1

    Returns
    -------
    `Data.FitsData`
        The input ``stokesData`` is modified in place but is also returned for convenience
    """
    if 'ISSTOKES' not in stokesData.headers[0] or not stokesData.headers[0]['ISSTOKES']:
        raise ValueError('{} is not demodulated Stokes data'.format(stokesData))

    if align3_cal['transform']['type'] != 'affine':
        raise ValueError("Given transform is not Affine ({}). This shouldn't be possible".format(
                align3_cal['transform']['type']))

    transpars = align3_cal['transform']
    transform = skit.AffineTransform(scale=[transpars['scale_x'], transpars['scale_y']],
                                     rotation=transpars['rotation'], shear=transpars['shear'],
                                     translation=[transpars['translation_x'], transpars['translation_y']])

    stokes_I = stokesData.data[0]
    for i in range(len(stokesData.data)):
        print('{}: Combining beams for Stokes {}'.format(tag(), stokesData.data_hdus[i].header['STOKES']))
        if i == 0:
            normed_data = stokesData.data[i]
        else:
            print('{}:  normalizing by Stokes I'.format(tag()))
            normed_data = stokesData.data[i] / stokes_I
        beam1, beam2 = Align3Calibration.split_beams(normed_data)

        beam2_trans = skit.warp(beam2.astype(np.float64), transform.inverse,
                                preserve_range=True, output_shape=beam1.shape, cval=np.nan)
        stokesData.data[i] = beam1 + beam2_trans

    print('{}: De-normalizing combined Stokes data'.format(tag()))
    combined_stokes_I = stokesData.data[0]
    for dh in stokesData.data_hdus[1:]:
        dh.data *= combined_stokes_I

    stokesData.headers[0]['HISTORY'] = 'Combined beams with ALIGN3 on {}'.format(time.asctime())

    return stokesData
    



    [docs]def combine_intensity_beams(fitsData, align3_cal):
    """Transform Cryo beam 2 into beam 1 coordinates and then add the beams together.

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Object containing demodulated Cryo data

    align3_cal : dict
        Dictionary with Affine transform parameters that map beam 2 -> beam 1

    Returns
    -------
    `Data.FitsData`
        The input ``fitsData`` is modified in place but is also returned for convenience
    """
    try:
        if fitsData.headers[0]['ISSTOKES']:
            raise ValueError('{} is demodulated Stokes data. Use `combine_polarimetric_beams` instead'.format(fitsData))
    except KeyError:
        pass

    if align3_cal['transform']['type'] != 'affine':
        raise ValueError("Given transform is not Affine ({}). This shouldn't be possible".format(
                align3_cal['transform']['type']))

    transpars = align3_cal['transform']
    transform = skit.AffineTransform(scale=[transpars['scale_x'], transpars['scale_y']],
                                     rotation=transpars['rotation'], shear=transpars['shear'],
                                     translation=[transpars['translation_x'], transpars['translation_y']])

    for i in range(len(fitsData.data)):
        print('{}: Combining beams for data HDU {}'.format(tag(), i))
        beam1, beam2 = Align3Calibration.split_beams(fitsData.data[i])

        beam2_trans = skit.warp(beam2.astype(np.float64), transform.inverse,
                                preserve_range=True, output_shape=beam1.shape, cval=np.nan)
        fitsData.data[i] = beam1 + beam2_trans

    fitsData.headers[0]['HISTORY'] = 'Combined beams with ALIGN3 on {}'.format(time.asctime())

    return fitsData
    



    [docs]def geometric_correction(fitsData, align3_cal, gain4_shifts):
    """Remove dispersion angle offset and spectral curvature from data

    Input data is assumed to be combined beam data and thus only beam 1 corrections are applied.

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Data to be corrected. Should already have both beams combined.

    align3_cal : dict
        Dictionary with ALIGN3 calibration, specifically the beam dispersion angles

    gain4_shifts : `Data.FitsData`
        Object containing Gain4 Calibration shifts

    Returns
    -------
    `Data.FitsData`
        The input data are modified in place, but they are also returned for convenience
    """
    rot_angle = -1 * align3_cal['angles']['beam1']
    shifts = gain4_shifts.data[0][0]  # These are the shifts for beam 1
    print('{}: Disp angle correction = {:.2f} deg'.format(tag(), rot_angle))
    for i in range(len(fitsData.data)):
        try:
            print('{}: Correcting geometry of Stokes {}'.format(tag(), fitsData.data_hdus[i].header['STOKES']))
        except KeyError:
            print('{}: Correcting geometry in data HDU {}'.format(tag(), i))

        data = fitsData.data[i]
        # We'll use this mask to re-nanify the final data after interpolation. It's a boolean mask
        mask = (~np.isfinite(data)).astype(np.int)
        data[np.isnan(data)] = np.nanmedian(data)

        print('{}:  rotating beam'.format(tag()))
        data_rot = spnd.interpolation.rotate(data, rot_angle, mode='nearest', reshape=False)
        mask_rot = spnd.interpolation.rotate(mask, rot_angle, mode='nearest', reshape=False)

        print('{}:  removing spectral curvature'.format(tag()))
        data_shift = Gain4Calibration.remove_sepectral_curvature(data_rot, shifts)
        mask_shift = Gain4Calibration.remove_sepectral_curvature(mask_rot, shifts)

        # It's very unlikely (impossible?) that a 1 in the OG mask to turn into 0, so we'll use the zeros as indications
        #  of where data are good.
        data_shift[mask_shift != 0] = np.nan

        fitsData.data[i] = data_shift

    fitsData.headers[0]['HISTORY'] = 'Removed rotation and spectral curvature on {}'.format(time.asctime())

    return fitsData
