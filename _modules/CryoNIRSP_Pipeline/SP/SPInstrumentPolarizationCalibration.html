    [docs]def main(raw_pol_dir, output, threads=1, linearize=True,
         cu_numbins_x=1, cu_numbins_y=1, demod_numbins_x=1, demod_numbins_y=1):

    pol_rdx_dir = '{}/polcal_proc_SP'.format(Data.TMPDIR)
    dmod1_name = '{}/DMOD_beam1.{}'.format(pol_rdx_dir, CONSTANTS['Data']['suffix'])
    dmod2_name = '{}/DMOD_beam2.{}'.format(pol_rdx_dir, CONSTANTS['Data']['suffix'])
    cmp1_name = '{}/CUpars_beam1.{}'.format(pol_rdx_dir, CONSTANTS['Data']['suffix'])
    cmp2_name = '{}/CUpars_beam2.{}'.format(pol_rdx_dir, CONSTANTS['Data']['suffix'])
    cmp1_resamp_name = '{}/CUpars_demodRCUH_beam1.{}'.format(pol_rdx_dir, CONSTANTS['Data']['suffix'])
    cmp2_resamp_name = '{}/CUpars_demodRCUH_beam2.{}'.format(pol_rdx_dir, CONSTANTS['Data']['suffix'])

    if linearize:
        lin_dir = os.path.join(raw_pol_dir, 'linearized')
        LinearityCorrection.main(raw_pol_dir, lin_dir, **CONSTANTS['LinearityKwargs'])
        raw_pol_dir = lin_dir

    if os.path.exists(dmod1_name) and os.path.exists(dmod2_name):
        print('{}: Found previous demodulation matrices in {}'.format(tag(), pol_rdx_dir))
        raw_demod = Data.DemodulationData()

        raw_demod.load_beam_from_file(dmod1_name, 1)
        print('{}: loaded {}'.format(tag(), dmod1_name))

        raw_demod.load_beam_from_file(dmod2_name, 2)
        print('{}: loaded {}'.format(tag(), dmod1_name))

    else:
        if not (os.path.exists(cmp1_resamp_name) and os.path.exists(cmp2_resamp_name)):
            if not (os.path.exists(cmp1_name) and os.path.exists(cmp2_name)):
                if not os.path.exists('{}/rdx_complete'.format(pol_rdx_dir)):
                    pol_args = {'raw_pol_dir': raw_pol_dir,
                                'threads': threads,
                                'cu_numbins_x': cu_numbins_x,
                                'cu_numbins_y': cu_numbins_y,
                                'demod_numbins_x': demod_numbins_x,
                                'demod_numbins_y': demod_numbins_y}
                    print('{}: Perfoming CryoNIRSP reduction on CS data in {}'.format(tag(), raw_pol_dir))
                    reduce_full_CS(**pol_args)

                print('{}: Using PA&C Modules to fit CU parameters'.format(tag()))
                run_PAC_CUpar(pol_rdx_dir, threads=threads)

            print('{}: Found raw fits to CU parameters in {}'.format(tag(), pol_rdx_dir))
            for base_name, resamp_name in zip([cmp1_name, cmp2_name],
                                              [cmp1_resamp_name, cmp2_resamp_name]):
                cuData = Data.FitsData(base_name, allow3D=True)
                print('{}: loaded {}'.format(tag(), base_name))
                print('{}: Resampling CU parameter fits to Demodulation RCUH'.format(tag()))
                resampData = pol_generic.resample_CU_params(cuData, demod_numbins_x, demod_numbins_y)
                resampData.write_out(resamp_name)

        print('{}: Found resampled fits to CU parameters in {}'.format(tag(), pol_rdx_dir))
        print('{}: Using PA&C Modules to compute demodulation matrices'.format(tag()))
        raw_demod = run_PAC_demod(pol_rdx_dir)

    print('{}: Generating Instrument Polarization Calibration'.format(tag()))
    instpol_calibration = generate_instrument_polarization_calibration(raw_demod, output)

    return instpol_calibration

    [docs]def reduce_full_CS(raw_pol_dir, threads=1, dark_cal='PolDarkCal', gain_cal='PolGainCal',
                   cu_numbins_x=1, cu_numbins_y=1, demod_numbins_x=1, demod_numbins_y=1):
    """Spawn the parallel reduction of an entire Calibration Sequence.

    This function sets up the right directories and default file names and then hands off the management to
    `manage_CS_rdx`.
    """
    if not (os.path.exists(raw_pol_dir) and os.path.isdir(raw_pol_dir)):
        raise FileNotFoundError('Cannot find raw PolCal directory {}'.format(raw_pol_dir))

    outdir = '{}/polcal_proc_SP'.format(Data.TMPDIR)
    os.makedirs(outdir, exist_ok=True)

    darkcal_file = generic.ensure_fits('{}/{}'.format(outdir, dark_cal))
    gaincal_file = generic.ensure_fits('{}/{}'.format(outdir, gain_cal))

    print('{}: Scanning polcal files in {}'.format(tag(), raw_pol_dir))
    RawDrawer = Data.RawDrawerData(raw_pol_dir)

    validate_inst_pol_task_data(RawDrawer)

    num_steps = RawDrawer.num_CS_steps
    print('{}: Found {} calibration sequence steps'.format(tag(), num_steps))

    pol_kwargs = pol_generic.gen_inst_pol_cal_objects(RawDrawer, darkcal_file, gaincal_file)
    pol_kwargs.update({'cu_numbins_x': cu_numbins_x,
                       'cu_numbins_y': cu_numbins_y,
                       'demod_numbins_x': demod_numbins_x,
                       'demod_numbins_y': demod_numbins_y})

    manage_kwargs = {'output_dir': outdir,
                     'threads': threads}

    manage_CS_rdx(RawDrawer, manage_kwargs, pol_kwargs)

    # Create a file that indicates all files were successfully reduced
    with open('{}/rdx_complete'.format(outdir), 'w') as f:
        f.write('CryoNIRSP PolCal reduction of Calibration sequence in {}\nsaved to {}/beam[12] on {}'. \
                format(raw_pol_dir, outdir, time.asctime()))
    


    [docs]def manage_CS_rdx(SoCCData, manage_kwargs, pol_kwargs):
    """Manage the parallel reduction of an entire Calibration Sequence.

    This function constructs a separate job for each CS step and then runs them such that there are never more jobs
    running than the requested number of threads.
    """
    num_steps = SoCCData.num_CS_steps
    numthread = int(manage_kwargs['threads'])
    print('{}: Running CryoNIRSP reduction on all files with {} threads'.format(tag(), numthread))

    # This loop sets up a list of mp processes. They have not been started.
    jobs = []
    pol_kwargs['dark_lock'] = mp.Lock()
    pol_kwargs['gain_lock'] = mp.Lock()
    outpre = manage_kwargs['output_dir']
    for j in range(num_steps):
        jobs.append(mp.Process(target=reduce_single_CS_step,
                               args=(SoCCData, j, outpre),
                               kwargs=pol_kwargs,
                               name='step_{:02}'.format(j)))

    # Now, start jobs and keep starting jobs until the max number of threads is reached
    #  Make a check every second to see if any jobs have finished and thus created room for new ones to start.
    i = 0
    while i < len(jobs):
        while [x.is_alive() for x in jobs].count(True) < numthread:
            jobs[i].start()
            i += 1
            if i >= len(jobs):
                break
        time.sleep(1)

    # Finally, make sure we wait for all jobs to complete before moving on.
    for p in jobs:
        p.join()

    return
    



    [docs]def reduce_single_CS_step(SoCCData, CS_step, outdir, dark_cal=None, gain_cal=None,
                          cu_numbins_x=1, cu_numbins_y=1, demod_numbins_x=1, demod_numbins_y=1,
                          dark_lock=NULLCONTEXT, gain_lock=NULLCONTEXT):
    """Do basic (i.e., dark and gain) reduction on data from a single CS step and then bin them for hand off to PAC

    Returns
    -------
    `Data.FitsData`
        Reduced and binned data for beam 1

    `Data.FitsData`
        Reduced and binned data for beam 2
    """
    polData = SoCCData.get_CS_step_data(CS_step)
    print('{}: Loaded CS step {} from {}'.format(tag(), CS_step, SoCCData))

    if dark_cal is not None:
        print('{}: Doing dark correction'.format(tag()))
        with dark_lock:
            print('{}: acquiring multiprocessing dark lock'.format(tag()))
            generic.dark_correction(polData, dark_cal)
            print('{}: releasing multiprocessing dark lock'.format(tag()))

    if gain_cal is not None:
        print('{}: Doing gain correction'.format(tag()))
        with gain_lock:
            print('{}: aqcuiring multiprocessing gain lock'.format(tag()))
            # It might seem a little weird to use unmodulated_gain_corr, but that's largely a naming issue.
            # The fact is that the PolCal GainCal does not preserve modulator states, which is why we use
            # the unmodulated gain correction.
            generic.unmodulated_gain_correction(polData, gain_cal)
            print('{}: releasing multiprocessing gain lock'.format(tag()))

    for xbin, ybin, suffix, name in zip([cu_numbins_x, demod_numbins_x],
                                        [cu_numbins_y, demod_numbins_y],
                                        ['cu_RCUH','demod_RCUH'],
                                        ['CU parameter fit', 'demodulation matrix calculation']):
        print('{}: extracting macropixels for {}'.format(tag(), name))
        binData1, binData2 = extract_macropixels(polData, xbin, ybin)

        rcuh_dir = '{}/{}'.format(outdir, suffix)
        print('{}: saving reduced PolCal Data for {} to {}'.format(tag(), name, rcuh_dir))
        write_reduced_CS_step(binData1, binData2, rcuh_dir, CS_step)

    return binData1, binData2
    


    [docs]def extract_macropixels(fitsData, numpix_x, numpix_y):
    """Bin data from each CryoNIRSP beam into macropixels suitable for hand-off to PAC modules

    The user specifies how many macropixels they want. Right now both beams are constrained to have the same final,
    macropixel shape, but this may change.

    If the requested number of macropixels do not fit evenly into the beams then some rows/columns may be lost. This
    probably doesn't matter too much, but be aware of it.

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Modulated CryoNIRSP data. This should have the full frame.

    numpix_x : int
        The number of spatial macropixels. This will be the same for each beam.

    numpix_y : int
        The number of spectral macropixels *in each beam*.

    Returns
    -------
    `Data.FitsData`
        Binned data for each modulator state of beam 1

    `Data.FitsData`
        Binned data for each modulator state of beam 2
    """

    nummod = fitsData.headers[0][S122['NumberOfModulatorStates']]
    beam1, beam2 = Align3Calibration.split_beams(fitsData._get_stacked_data())
    print('{}: full beam 1 shape: {}'.format(tag(), beam1.shape[:-1]))
    print('{}: full beam 2 shape: {}'.format(tag(), beam2.shape[:-1]))

    print('{}: both beams will be binned to ({}, {}, 1)'.format(tag(), numpix_x, numpix_y))

    beamData_list = []
    for i, beam in enumerate([beam1, beam2]):
        print('{}: binning beam {}'.format(tag(), i + 1))

        binned_data = pol_generic.downsample_data(beam, numpix_x, numpix_y)
        if binned_data.shape[2] != nummod:
            raise ValueError('Binned data does not have {} modulator states! ({})'.format(nummod, binned_data.shape[2]))

        binData = Data.FitsData(allow3D=True)
        binData.origin = 'extract macropixels'

        # Here we add the extra dummy dimension required by PA&C
        binData.init_with_ndarray(binned_data[:, :, None, :], autoflush=False)

        xbin = beam.shape[0] // numpix_x
        ybin = beam.shape[1] // numpix_y
        for j, h in enumerate(binData.hdulist):
            h.header = fitsData.hdulist[j].header.copy()
            h.header[S122['SoftwareBinningX']] = xbin
            h.header[S122['SoftwareBinningY']] = ybin
            h.header['OGX'] = beam.shape[0]
            h.header['OGY'] = beam.shape[1]
            h.header['CRSP_BEM'] = i + 1

        beamData_list.append(binData)

    return beamData_list[0], beamData_list[1]

    [docs]def run_PAC_CUpar(run_dir, threads=1, overwrite=False):
    """Spawn the PAC Modules' CU Parameter fit code
    """
    from PAC_Pipeline import FitCUParams
    from PAC_Pipeline import generic as PAC_generic
    telescope_db = PAC_generic.get_default_telescope_db()

    jobs = []
    pac_keywords = {'overwrite': overwrite,
                    'save_stats': True,
                    'telescope_db': telescope_db,
                    'threads': threads // 2 or 1,
                    'suffix': CONSTANTS['Data']['suffix']}

    for i in [1, 2]:
        beam_dir = '{}/cu_RCUH/beam{}'.format(run_dir, i)
        beam_output = '{}/CUpars_beam{}.{}'.format(run_dir, i, CONSTANTS['Data']['suffix'])
        pac_keywords['super_name'] = 'Beam{}'.format(i)
        jobs.append(mp.Process(target=FitCUParams.main,
                               args=[beam_dir, beam_output],
                               kwargs=pac_keywords,
                               name='Beam{}'.format(i)))

    i = 0
    while i < len(jobs):
        while [x.is_alive() for x in jobs].count(True) < threads:
            jobs[i].start()
            i += 1
            if i >= len(jobs):
                break
        time.sleep(1)

    for p in jobs:
        p.join()

    return

    """Spawn the PAC Modules' Demodulation matrix calculation fit code and collect the resulting demodulation matrices
    """
    from PAC_Pipeline import GenerateDemodMatrices
    from PAC_Pipeline import generic as PAC_generic

    #TODO: Make a keyword param
    threads = 2
    jobs = []
    pac_kwargs = {'overwrite': overwrite,
                  'suffix': CONSTANTS['Data']['suffix'],
                  'telescope_db': PAC_generic.get_default_telescope_db()}

    for i in [1, 2]:
        beam_dir = '{}/demod_RCUH/beam{}'.format(run_dir, i)
        CUpar_file = '{}/CUpars_demodRCUH_beam{}.{}'.format(run_dir, i, CONSTANTS['Data']['suffix'])
        beam_output = '{}/DMOD_beam{}.{}'.format(run_dir, i, CONSTANTS['Data']['suffix'])
        jobs.append(mp.Process(target=GenerateDemodMatrices.main,
                               args=[beam_dir, CUpar_file, beam_output],
                               kwargs=pac_kwargs,
                               name='Beam{}'.format(i)))

    i = 0
    while i < len(jobs):
        while [x.is_alive() for x in jobs].count(True) < threads:
            jobs[i].start()
            i += 1
            if i >= len(jobs):
                break
        time.sleep(1)

    for p in jobs:
        p.join()

    demodData = Data.DemodulationData()
    for i in [1, 2]:
        demodData.load_beam_from_file('{}/DMOD_beam{}.{}'.format(run_dir, i, CONSTANTS['Data']['suffix']), i)

    return demodData
    


    [docs]def generate_instrument_polarization_calibration(dmodData, output):
    """Take the output of PAC's demod and upsample to the full Cryo frame

    Right now this is overly simplified but can change.

    Returns
    -------
    `Data.DemodulationData`
        Full-frame (i.e., both beams) demodulation matrices, contained only in "beam1" of this object
    """
    print('{}: extracting demoulation matrices'.format(tag()))
    numx1, numy1, numl1, nums1, numm1 = dmodData.beam1.data.shape
    numx2, numy2, numl2, nums2, numm2 = dmodData.beam2.data.shape

    if numl1 != 1 or numl2 != 1:
        raise ValueError("The PA&C dummy dimension is > 1!")

    if nums1 != 4 or nums2 != 4:
        raise ValueError("The number of Stokes parameters ({}) is not 4. This shouldn't be possible".format(nums1))

    if numm1 != numm2:
        raise ValueError("Beams 1 and 2 do not have the same number of modulation states! ({} vs {})".format(numm1,
                                                                                                             numm2))

    OG_shape_1 = (dmodData.beam1.header['OGX'], dmodData.beam1.header['OGY'])
    OG_shape_2 = (dmodData.beam2.header['OGX'], dmodData.beam2.header['OGY'])
    print('{}: original full-frame shape for beam 1: {}'.format(tag(), OG_shape_1))
    print('{}: original full-frame shape for beam 2: {}'.format(tag(), OG_shape_2))

    if OG_shape_1[0] != OG_shape_2[0]:
        raise ValueError("Beams 1 and 2 do not have the same original spatial size ({} vs {})".format(OG_shape_1[0],
                                                                                                      OG_shape_2[0]))
    big_beam_1 = pol_generic.upsample_data(dmodData.beam1.data, *OG_shape_1)
    big_beam_2 = pol_generic.upsample_data(dmodData.beam2.data, *OG_shape_2)
    big_dmod = np.concatenate((big_beam_1, big_beam_2), axis=1)

    # Here we drop the dummy dimension required by PA&C (numl in this case)
    big_dmod = big_dmod[:, :, 0, :, :]
    print('{}: final shape of combined beam full-frame demodulation matrices: {}'.format(tag(), big_dmod.shape))

    final_dmod = Data.DemodulationData()
    final_dmod.init_beam_with_ndarray(big_dmod, 1)
    final_dmod.data_hdus[0].header = dmodData.beam1.header.copy()

    final_dmod.write_out(output, overwrite=True)

    return final_dmod
