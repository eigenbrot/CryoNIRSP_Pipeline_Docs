    [docs]def resample_CU_params(cuData, numpix_x, numpix_y):
    """Resample the CU fit parameters to a different RCUH in preparation for demodulation matrix calculations

    Right now this is designed primarily to upsample data. It will work if the destination RCUH is smaller than the
    source, but the method used will hurt the accuracy of downsampling.

    Parameters
    ----------
    cuData : `Data.FitsData`
        CU parameter file object. A little strange in the sense that it has a single HDU with data shape (x, y, 1, Z),
        where Z is the number of fit parameters.

    numpix_x : int
        Target spatial size of upsampled data

    numpix_y : int
        Target spectral size of upsampled data

    Returns
    -------
    `Data.FitsData`
        Object with a single HDU with shape (numpix_x, numpix_y, 1, Z)
    """
    data = cuData.data[0]
    resampled_data = upsample_data(data, numpix_x, numpix_y)

    phdu = pyfits.PrimaryHDU(resampled_data, header=cuData.headers[0])

    full_x = cuData.headers[0]['OGX']
    full_y = cuData.headers[0]['OGY']
    xbin = full_x / numpix_x
    ybin = full_y / numpix_y

    print('{}: resampled spatial binning compared to full-frame = {:.2f}'.format(tag(), xbin))
    print('{}: resampled spectral binning compared to full-frame = {:.2f}'.format(tag(), ybin))

    phdu.header[S122['SoftwareBinningX']] = xbin
    phdu.header[S122['SoftwareBinningY']] = ybin

    resampData = Data.FitsData(allow3D=True)
    resampData.origin = 'resample CU params'
    resampData.hdulist = [phdu]

    return resampData
    



    [docs]def upsample_data(data, big_x, big_y):
    """Helper-esq function that upsamples data via nearest-neighbor interpolation.

    The upsampling is done on the first two dimensions only, all other dimensions are preserved in output.
    Because interpolation is used, non-integer up-binning can be done no problem. Also, technically this function can
    downsample, but accuracy is not guaranteed.

    Data must be at least 3D.

    Parameters
    ----------
    data : `numpy.ndarray`
        Data with shape (x, y, m1, ..., mn)

    big_x : int
        Output size in 0th dimension

    big_y : int
        Output size in 1st dimension

    Returns
    -------
    `numpy.ndarray`
        Downsampled data with shape (big_x, big_y, m1, ..., mn)
    """
    numx, numy = data.shape[:2]
    uninterp_shape = data.shape[2:]
    print('{}: macropixeled shape: {}'.format(tag(), data.shape))
    print('{}: upsampling data to {}'.format(tag(), (big_x, big_y) + uninterp_shape))

    x = np.arange(big_x / 2, big_x * numx, big_x) / numx
    y = np.arange(big_y / 2, big_y * numy, big_y) / numy
    xi, yi = np.mgrid[0:big_x:big_x * 1j, 0:big_y:big_y * 1j]

    flat_size = np.prod(uninterp_shape)
    flat_data = np.reshape(data, (numx, numy, flat_size))
    flat_output = np.zeros((big_x, big_y, flat_size))
    print('{}: interpolating'.format(tag()))
    for i in range(flat_size):
        print('{}:  position {}'.format(tag(), np.unravel_index(i, uninterp_shape)))
        flat_output[:, :, i] = spi.interpn((x, y), flat_data[:, :, i], (xi, yi), bounds_error=False,
                                           fill_value=None, method='nearest')

    full_output = np.reshape(flat_output, (big_x, big_y) + uninterp_shape)

    return full_output
    



    [docs]def generate_polcal_dark_calibration(RawDrawer):
    """Find the dark frames in the Calibration Sequence (CS) and average them together

    It is assumed that all Calibration Sequences have 2 darks. Preferably one at the start and one at the end.

    Parameters
    ----------
    RawDrawer : `Data.RawDrawerData`
        Object containing data from a PolCal run

    Returns
    -------
    `Data.FitsData`
        Data object of first dark

    `Data.FitsData`
        Data object of second dark

    `Data.FitsData`
        Data object of average dark
    """
    dark_file_list = RawDrawer.get_dark_file_list()
    if len(dark_file_list) != 2:
        raise ValueError('PolCal Calibration Sequence needs exactly 2 dark steps (found {})'.format(len(dark_file_list)))

    output_dark_list = []
    for i in range(2):
        print('{}: Working on dark step #{}'.format(tag(), i+1))
        darkData = Data.FitsData()
        darkData.load_from_filelist(dark_file_list[i])

        print('{}: Averaging data'.format(tag()))
        avgDark = generic.average_FitsData(darkData)

        print('{}: Computing std of data'.format(tag()))
        stdDark = generic.stddev_FitsData(darkData, avgDark)
        avgDark.set_unc(stdDark)
        output_dark_list.append(avgDark)

    print('{}: Computing total average'.format(tag()))
    all_darkData = Data.FitsData()
    all_darkData.load_from_filelist(dark_file_list[0] + dark_file_list[1])
    all_avgData = generic.average_FitsData(all_darkData)
    all_stdData = generic.stddev_FitsData(all_darkData, all_avgData)
    all_avgData.set_unc(all_stdData)

    return output_dark_list[0], output_dark_list[1], all_avgData
    



    [docs]def generate_polcal_gain_calibration(RawDrawer, PolDarkCal1, PolDarkCal2, percentile=0.95):
    """Find the clear frames in the Calibration Sequence (CS) and average them together

    It is assumed that all Calibration Sequences have 2 clear. Preferably one at the start and one at the end.

    Parameters
    ----------
    RawDrawer : `Data.RawDrawerData`
        Object containing data from a PolCal run

    PolDarkCal1 : `Data.FitsData`
        First Dark step object

    PolDarkCal2 : `Data.FitsData`
        Second Dark step object

    percentile : float
        Percentage of clear flux CDF to use for clear normalization

    Returns
    -------
    `Data.FitsData`
        Data object of average clear frame
    """
    clear_file_list = RawDrawer.get_clear_file_list()
    if len(clear_file_list) != 2:
        raise ValueError('PolCal Calibration Sequence needs exactly 2 clear steps (found {})'.format(len(clear_file_list)))

    dark_list = [PolDarkCal1, PolDarkCal2]
    clear_data_list = []
    for i in range(2):
        print('{}: Working on clear step {}'.format(tag(), i+1))
        clearData = Data.FitsData()
        clearData.load_from_filelist(clear_file_list[i])

        print('{}: Averaging data'.format(tag()))
        avgClear = generic.average_modulated_FitsData(clearData)

        print('{}: Computing std of data'.format(tag()))
        stdGain = generic.stddev_modulated_FitsData(clearData, avgClear)
        avgClear.set_unc(stdGain)

        print('{}: Subtracting dark'.format(tag()))
        generic.dark_correction(avgClear, dark_list[i])
        clear_data_list.append(avgClear)

    print('{}: Computing total average'.format(tag()))
    all_clearData = Data.concatenate_fitsData(clear_data_list[0], clear_data_list[1])
    all_avgData = generic.average_FitsData(all_clearData)

    print('{}: Computing normalization at {} percentile'.format(tag(), percentile*100))
    # Now normalize by whatever percentile was requested
    cdf = np.sort(np.ravel(all_avgData.data[0]))
    perc95 = int(np.round(cdf.size * percentile))
    norm = cdf[perc95]

    print('{:}: Normalization = {:.4f}. Applying to data'.format(tag(), norm))
    all_avgData.data[0] = all_avgData.data[0] / norm
    all_avgData.unc[0] = all_avgData.unc[0] / norm

    return all_avgData
    



    [docs]def gen_inst_pol_cal_objects(RawDrawer, dark_cal_file, gain_cal_file):
    """Manage the creation of auxiliary PolCal calibration objects. Namely darks and gains.

    If the requested files are found they are simply loaded, otherwise they are generated with the PolCal data
    """
    # Dark
    ######
    dark_prefix = dark_cal_file.split('.{}'.format(CONSTANTS['Data']['suffix']))[0]
    dark1_filename = '{}_1.{}'.format(dark_prefix, CONSTANTS['Data']['suffix'])
    dark2_filename = '{}_2.{}'.format(dark_prefix, CONSTANTS['Data']['suffix'])
    if not (os.path.exists(dark_cal_file) and os.path.exists(dark1_filename) and os.path.exists(dark2_filename)):
        print('{}: Could not find dark cal objects, attempting to generate'.format(tag()))
        DarkCal1, DarkCal2, DarkCalTot = generate_polcal_dark_calibration(RawDrawer)
        DarkCal1.write_out(dark1_filename, overwrite=True)
        DarkCal2.write_out(dark2_filename, overwrite=True)
        DarkCalTot.write_out(dark_cal_file, overwrite=True)
    else:
        DarkCal1 = Data.FitsData(dark1_filename)
        print('{}: Loaded {}'.format(tag(), dark1_filename))
        DarkCal2 = Data.FitsData(dark2_filename)
        print('{}: Loaded {}'.format(tag(), dark2_filename))
        DarkCalTot = Data.FitsData(dark_cal_file)
        print('{}: Loaded {}'.format(tag(), dark_cal_file))

    # Gain
    ######
    if not os.path.exists(gain_cal_file):
        print('{}: Could not find {}, attempting to generate'.format(tag(), gain_cal_file))
        GainCal = generate_polcal_gain_calibration(RawDrawer, DarkCal1, DarkCal2)
        GainCal.write_out(gain_cal_file, overwrite=True)
    else:
        GainCal = Data.FitsData(gain_cal_file)
        print('{}: Loaded {}'.format(tag(), gain_cal_file))

    pol_kwargs = {'dark_cal': DarkCalTot,
                  'gain_cal': GainCal}

    return pol_kwargs
    



    [docs]def downsample_data(data, small_x, small_y):
    """Helper-esq function that downsamples a data array by averaging all values in each bin.

    The downsampling is done over the first two dimensions only, but all other dimensions are preserved on output.

    Data must be at least 3D

    Parameters
    ----------
    data : `numpy.ndarray`
        Data with shape (x, y, m1, ..., mn)

    small_x : int
        Output size in 0th dimension

    small_y : int
        Output size in 1st dimension

    Returns
    -------
    `numpy.ndarray`
        Downsampled data with shape (small_x, small_y, m1, ..., mn)
    """
    if small_x > data.shape[0] or small_y > data.shape[1]:
        raise IndexError('Cannot bin to more pixels than input')

    xbin = data.shape[0] // small_x
    ybin = data.shape[1] // small_y

    print('{}: spatial binning: {}'.format(tag(), xbin))
    print('{}: spectral binning: {}'.format(tag(), ybin))

    # The "or None" happens if the math = 0. The None is needed so *end works in slices as expected below
    xend = small_x * xbin - data.shape[0] or None
    yend = small_y * ybin - data.shape[1] or None

    if (data.shape[0] / small_x) % 1 != 0:
        print('{}: Spatial binning amount does not exactly fill input dimension. '
              '{} row[s] will be lost'.format(tag(), abs(xend)))

    if (data.shape[1] / small_y) % 1 != 0:
        print('{}: Spectral binning amount does not exactly fill input dimension. '
              '{} column[s] will be lost'.format(tag(), abs(yend)))

    numx, numy = data.shape[:2]
    uninterp_shape = data.shape[2:]
    flat_size = np.prod(uninterp_shape)
    flat_data = np.reshape(data, (numx, numy, flat_size))

    x_binned = np.zeros((small_x, numy, flat_size))
    macro_px = np.zeros((small_x, small_y, flat_size))

    for x in range(xbin):
        x_binned += flat_data[x:xend:xbin, :, :]
    for y in range(ybin):
        macro_px += x_binned[:, y:yend:ybin, :]

    macro_px /= xbin * ybin

    return np.reshape(macro_px, (small_x, small_y) + uninterp_shape)
