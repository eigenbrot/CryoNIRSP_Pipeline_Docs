    [docs]def main(gain_dir, outpre, dark_cal=None):

    gainData = Data.GainData(gain_dir, dataset_ID='Gain1')

    validate_gain1_task(gainData)

    print('{}: Averaging data'.format(tag()))
    avgData = generic.average_modulated_FitsData(gainData)

    print('{}: Computing stddev'.format(tag()))
    stdData = generic.stddev_modulated_FitsData(gainData, avgData)
    avgData.set_unc(stdData)

    if dark_cal is not None:
        print('{}: Removing dark signal'.format(tag()))
        generic.dark_correction(avgData, dark_cal)

    validate_gain1_calibration(avgData)

    print('{}: Generating Gain1 Calibration'.format(tag()))
    gain_calibration = generate_gain1_calibration(avgData, outpre, gain_dir)

    return gain_calibration

    [docs]def validate_gain1_calibration(avgGain):

    for i in range(len(avgGain.data)):
        if np.min(avgGain.data[i]) < 0:
            raise ValueError('Gain1 Calibration has negative values')

        std_frac = np.mean(avgGain.data[i] / avgGain.unc[i])
        if std_frac < 1:
            raise ValueError('Gain1 Calibration stddev is too high ({:.3f} of mean)'.format(std_frac))
    



    [docs]def generate_gain1_calibration(avgData, outpre, gain_dir):
    outname = outpre + '.fits'
    avgData.origin = 'Gain1 Calibration'
    avgData.headers[0]['HISTORY'] = 'Generated with data from "{}" on {}'.format(os.path.abspath(gain_dir),
                                                                                 time.asctime())
    avgData.write_out(outname, overwrite=True)

    return avgData
