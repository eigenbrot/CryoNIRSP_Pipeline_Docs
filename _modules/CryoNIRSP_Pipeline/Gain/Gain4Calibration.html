    [docs]def main(gain_dir, align3_cal, outpre, dark_cal=None, lamp_gain_cal=None, linearize=True, debug=False):

    if linearize:
        CONSTANTS['LinearityKwargs']['threshold_file'] = None
        lin_dir = os.path.join(gain_dir, 'linearized')
        LinearityCorrection.main(gain_dir, lin_dir, **CONSTANTS['LinearityKwargs'])
        gain_dir = lin_dir

    print('{}: Loading data from {}'.format(tag(), gain_dir))
    gainData = Data.Gain4Data(gain_dir)

    validate_gain4_task(gainData)

    print('{}: Averaging data'.format(tag()))
    avgData = generic.average_FitsData(gainData)

    print('{}: Computing stddev'.format(tag()))
    stdData = generic.stddev_FitsData(gainData, avgData)
    avgData.set_unc(stdData)

    if dark_cal is not None:
        print('{}: Removing dark signal'.format(tag()))
        generic.dark_correction(avgData, dark_cal)

    if lamp_gain_cal is not None:
        print('{}: Applying lamp gain correction'.format(tag()))
        generic.modulated_gain_correction(avgData, lamp_gain_cal)

    print('{}: Extracting spectrograph beams'.format(tag()))
    beam1, beam2 = extract_detilted_beams(avgData, align3_cal)
    if debug:
        pyfits.PrimaryHDU(beam1).writeto('gain4_beam1.fits', overwrite=True)
        pyfits.PrimaryHDU(beam2).writeto('gain4_beam2.fits', overwrite=True)
        print("{}: Saved split, detilted beams as 'gain4_beam[12].fits'".format(tag()))

    final_beam_list = []
    final_shift_list = []
    for i, b in enumerate([beam1, beam2]):
        i += 1  # Because it's beams 1 and 2

        print('{}: BEAM {} - Computing spectral curvature'.format(tag(), i))
        shifts = find_spectral_curvature(b)
        final_shift_list.append(shifts)

        print('{}: BEAM {} - Removing spectral curvature'.format(tag(), i))
        deshifted = remove_sepectral_curvature(b, shifts)
        if debug:
            pyfits.PrimaryHDU(deshifted).writeto('gain4_deshifted{}.fits'.format(i))
            print("{}: Saved deshifted image to 'gain4_deshifted{}.fits'".format(tag(), i))

        print('{}: BEAM {} - Computing median spectrum'.format(tag(), i))
        avg_spec = compute_average_spectrum(deshifted)

        print('{}: BEAM {} - Constructing full beam average spectra'.format(tag(), i))
        full_avg = generate_average_spectra(avg_spec, shifts)
        if debug:
            pyfits.PrimaryHDU(full_avg).writeto('gain4_avgspec{}.fits'.format(i), overwrite=True)
            print("{}: Saved full beam average spectra to 'gain4_avgspec{}.fits'".format(tag(), i))

        final_beam_list.append(full_avg)

    print('{}: Combining spectrograph beams'.format(tag()))
    big_avg = combine_retilted_beams(final_beam_list[0], final_beam_list[1], align3_cal)

    if debug:
        pyfits.PrimaryHDU(big_avg).writeto('gain4_bigavg.fits', overwrite=True)
        print("{}: Saved restitched/tilted full average to 'gain4_bigavg.fits'".format(tag()))

    modAvg = generic.average_modulated_FitsData(gainData)
    modAvg.set_unc(generic.stddev_modulated_FitsData(gainData))
    if dark_cal is not None:
        generic.dark_correction(modAvg, dark_cal)

    print('{}: Removing average solar spectra'.format(tag()))
    remove_solar_spectra(modAvg, big_avg)

    print('{}: Normalizing Gain 4'.format(tag()))
    generic.normalize_modulated_FitsData(modAvg)

    validate_gain4_calibration(modAvg)

    gainCal, gain_shifts = generate_gain4_calibration(modAvg, final_shift_list, outpre, gain_dir)

    return gainCal, gain_shifts

    [docs]def remove_solar_spectra(gainData, full_chip_average):
    """Remove the Solar spectrum from the original data

    Parameters
    ----------
    gainData : `Data.FitsData`
        The average Solar Gain data used to measure the solar spectrum

    full_chip_average : `numpy.ndarray`
        The average solar spectrum, reshifted and tilted to match the original data
   """

    for i in range(len(gainData.data)):
        gainData.data[i] = gainData.data[i] / full_chip_average
        if gainData.unc is not None:
            gainData.unc[i] = gainData.unc[i] / full_chip_average
    


    [docs]def generate_gain4_calibration(gainData, shifts, outpre, gain_dir):
    """
    gainData : `Data.FitsData`
        The average Solar Gain data used to measure the solar spectrum, with solar spectra removed

    shifts : list
        List of 2 `numpy.ndarray` that contain the spectral shifts for beams 1 and 2, in that order.

    outpre : str
        Prefix to use for output files

    gain_dir : str
        The location of the data that were used to make this Cal object

    Returns
    -------
    `Data.FitsData`
        A Gain4 Calibration object

    `Data.FitsData`
        The spectra shifts for each beam
    """
    outname = outpre + '.' + CONSTANTS['Data']['suffix']
    gainData.origin = 'Gain 4 Calibration'
    gainData.headers[0]['HISTORY'] = 'Generated with data from "{}" on {}'.format(os.path.abspath(gain_dir),
                                                                                  time.asctime())
    gainData.write_out(outname, overwrite=True)

    dummy_primary = pyfits.PrimaryHDU(header=gainData.headers[0].copy())
    shift_hdu = pyfits.ImageHDU(np.vstack(shifts))
    shift_hdu.header['EXTNAME'] = 'Spectral shifts'

    shiftData = Data.FitsData()
    shiftData.origin = 'Gain 4 shifts'
    shiftData.hdulist = pyfits.HDUList([dummy_primary, shift_hdu])
    shift_outname = '{}_shifts.{}'.format(outpre, CONSTANTS['Data']['suffix'])
    shiftData.write_out(shift_outname, overwrite=True)

    return gainData, shiftData
    


    [docs]def find_spectral_curvature(data, median_spec_window=5, min_corr=0., peak_interp_width=5, max_shift=30,
                            edge_chop=50, debug=False):
    """Compute the spectral shift of each spatial channel relative to a central spectrum

    The reference spectrum is chosen to be as close to the middle of the spectral range as possible, while ensuring the
    quality of the spectrum is good (i.e., no NaN's, etc.). Each spatial pixel is then correlated with this reference
    spectrum and the correlation peak is interpolated to find the shift for each position.

    After all shifts are computed they are moved to a global mean of 0 so that effects of shifting spectra off the
    detector are minimized.

    Parameters
    ----------
    data : `numpy.ndarray`
        A 2D set of spectra with shape (X, Y). Should be aligned so the dispersion axis lies along array axes

    median_spec_window : int
        Window (in +/- pixels) around the centerr of the slit to use in determining the reference spectrum

    min_corr : float
        The minimum correlation amount required for a measured shift to be considered "good"

    peak_interp_width : int
        The number of points (+/-) around each correlation peak to use when interpolating the peak position. Directly
        passed to `peakutil.interpolate` as ``width``

    max_shift : float
        Any measured shifts that are larger than this are deemed suspicious and thus "bad"

    edge_chop : int
        Amount to chop off each end of the spectral dimension prior to correlation. This is very helpful in cases
        where the spectral edges are low quality or signal.

    debug : bool
        If True generate a bunch of useful FITS files showing intermediate steps

    Returns
    -------
    `numpy.ndarray`
        Array of length X, containing the spectral shift of each spatial location
    """
    numspec = data.shape[0]
    refspec = np.nanmedian(data[numspec // 2 - median_spec_window:numspec // 2 + median_spec_window, :], axis=0)
    refspec -= np.nanmean(refspec)
    refspec = refspec[edge_chop:-edge_chop]

    shifts = np.zeros(numspec)
    good_idx = []
    bad_idx = []

    for i in range(numspec):
        spec = data[i, :] - np.nanmean(data[i, :])
        spec = spec[edge_chop:-edge_chop]

        corr = np.correlate(spec, refspec, mode='same')
        idx = pku.indexes(corr, min_dist=corr.size)

        if idx - corr.size // 2 < max_shift and corr[idx] >= min_corr:
            shifts[i] = pku.interpolate(np.arange(corr.size), corr, ind=idx, width=peak_interp_width)[0] - corr.size // 2
            good_idx.append(i)
        else:
            shifts[i] = np.nan
            bad_idx.append(i)

    good_idx = np.array(good_idx)
    bad_idx = np.array(bad_idx)
    if bad_idx.size > 0:
        shifts[bad_idx] = np.interp(bad_idx, good_idx, shifts[good_idx])

    shifts -= np.mean(shifts)

    if debug and MPLT:
        fig = plt.figure()
        ax = fig.add_subplot(111)
        ax.set_xlabel('pixel')
        ax.set_ylabel('shift')
        x = np.arange(numspec)
        ax.plot(x[good_idx], shifts[good_idx], 'k-', lw=0.5)
        if bad_idx.size > 0:
            ax.plot(x[bad_idx], shifts[bad_idx], 'g.')
        ax.set_ylim(max(-2 * max_shift, 1.2 * min(shifts[good_idx])),
                    min(2 * max_shift, 1.2 * max(shifts[good_idx])))
        fig.savefig('spectral_shifts.pdf')
        plt.close(fig)

    return shifts
    


    [docs]def remove_sepectral_curvature(data, shifts):
    """Shift the spectrum of each spatial position by the amount specified

    Parameters
    ----------
    data : `numpy.ndarray`
        2D spectra to be shifted. Shape (X, Y)

    shifts : `numpy.ndarray`
        Array of length X, containing the spectral shift of each spatial location

    Returns
    -------
    `numpy.ndarray`
        Array of shifted spectra. Same shape as input.
    """
    numspec = data.shape[0]

    output = np.zeros(data.shape)
    for i in range(numspec):
        output[i, :] = spnd.interpolation.shift(data[i, :], -shifts[i], mode='reflect')

    return output
    


    [docs]def compute_average_spectrum(spectra, norm_frac=0.6):
    """Use information from all spatial positions to compute an average spectrum

    Prior to averaging all spatial positions are normalized by a fraction of their total flux. The final, averaged
    spectrum is normalized to have a max of 1.

    Parameters
    ----------
    spectra : `numpy.ndarray`
        2D array of shape (X, Y) containing Cryo spectra. Should probably have have its spectral curvature removed.

    norm_frac : float
        The fraction of each spectrum's total flux to use for normalization. In other words, a CDF is computed for each
        spectrum and the value ``norm_frac`` * CDF[-1] is used for normalization.

    Returns
    -------
    `numpy.ndarray`
        Single averaged spectrum. Shape (Y,)
    """
    continuum_levels = np.sort(spectra, axis=1)[:, int(spectra.shape[1] * norm_frac)]
    normd_spec = spectra / continuum_levels[:, None]

    avg_spec = np.nanmedian(normd_spec, axis=0)
    avg_spec /= np.max(avg_spec)

    return avg_spec
    


    [docs]def generate_average_spectra(avg_spectrum, shifts):
    """Fill the full, 2D Cryo beam with the average spectrum

    Basically we just reapply the spectral curvature to the averaged spectrum

    Parameters
    ----------
    avg_spectrum : `numpy.ndarray`
        1D array of length Y containing the average spectrum

    shifts : `numpy.ndarray`
        Array of length X containing the spectral shift of each spatial location

    Returns
    -------
    `numpy.ndarray`
        Array with shape (X, Y). The average spectrum shifted to the correct spectral location at each spatial location.
    """
    numspec = shifts.size
    output = np.zeros((numspec, avg_spectrum.size))

    for i in range(numspec):
        output[i] = spnd.interpolation.shift(avg_spectrum, shifts[i], mode='reflect')

    return output
    



    [docs]def extract_detilted_beams(fitsData, geometric_cal):
    """Split full-frame SP data along the beam edge and rotate each beam by the dispersion angle

    The initial split of the beams relies values in CryoNIRSP_Pipeline.data.ALIGN3_NUMBERS

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Full-frame Cryo data. Only the first data HDU will be used

    geometric_cal : dict
        Dictionary containing the dispersion angle of each beam

    Returns
    -------
    `numpy.ndarray`
        De-tilted beam 1 array
    `numpy.ndarray`
        De-tilted beam 2 array
    """
    data = fitsData.data[0]
    beam1, beam2 = Align3Calibration.split_beams(data)

    beam1[np.isnan(beam1)] = np.nanmedian(beam1)
    beam2[np.isnan(beam2)] = np.nanmedian(beam2)

    theta_1 = -1 * geometric_cal['angles']['beam1']
    theta_2 = -1 * geometric_cal['angles']['beam2']

    print('{}: Rotating beam 1 by {:.2f} deg'.format(tag(), theta_1))
    beam1_rot = spnd.interpolation.rotate(beam1, theta_1, mode='nearest', reshape=False)
    print('{}: Rotating beam 2 by {:.2f} deg'.format(tag(), theta_2))
    beam2_rot = spnd.interpolation.rotate(beam2, theta_2, mode='nearest', reshape=False)

    return beam1_rot, beam2_rot
    



    [docs]def combine_retilted_beams(beam1_rot, beam2_rot, geometric_cal):
    """Re-stitch two rotated beams back to their original, full-frame locations.

    The beams are first rotated back to their original angles and then simply mashed together.

    Parameters
    ----------
    beam1_rot : `numpy.ndarray`
        Array containing rotated beam 1. Shape (X, Y1)

    beam2_rot : `numpy.ndarray`
        Array containing rotated beam 2. Shape (X, Y2)

    geometric_cal : dict
        Dictionary containing the dispersion angle of each beam

    Returns
    -------
    `numpy.ndarray`
        Array of shape (X, Y1 + Y2) containing re-rotated, stitched beams.
    """
    if beam1_rot.shape[0] != beam2_rot.shape[0]:
        raise ValueError('Beams 1 and 2 do not share the same first dimension ({} vs {})'.format(beam1_rot.shape[0],
                                                                                                 beam2_rot.shape[0]))

    theta_1 = geometric_cal['angles']['beam1']
    theta_2 = geometric_cal['angles']['beam2']

    print('{}: Rotating beam 1 by {:.2f} deg'.format(tag(), theta_1))
    beam1 = spnd.interpolation.rotate(beam1_rot, theta_1, mode='nearest', reshape=False)
    print('{}: Rotating beam 2 by {:.2f} deg'.format(tag(), theta_2))
    beam2 = spnd.interpolation.rotate(beam2_rot, theta_2, mode='nearest', reshape=False)

    return np.hstack((beam1, beam2))
