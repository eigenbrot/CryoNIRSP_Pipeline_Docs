    [docs]def main(raw_pol_dir, output, threads=1):

    pol_rdx_dir = '{}/proc'.format(raw_pol_dir)
    dmod1_name = '{}/DMOD_beam1.fits'.format(pol_rdx_dir)
    cmp1_name = '{}/CUpars_beam1.fits'.format(pol_rdx_dir)

    if os.path.exists(dmod1_name):
        print('{}: Found previous demodulation matrices in {}'.format(tag(), pol_rdx_dir))
        raw_demod = Data.DemodulationData()
        raw_demod.load_beam_from_file(dmod1_name, 1)
        print('{}: loaded {}'.format(tag(), dmod1_name))

    else:
        if os.path.exists(cmp1_name):
            print('{}: Found previous fit to CU parameters in {}'.format(tag(), pol_rdx_dir))
        else:
            if not os.path.exists('{}/rdx_complete'.format(pol_rdx_dir)):
                pol_args = {'raw_pol_dir': raw_pol_dir,
                            'threads': threads}
                print('{}: Perfoming CryoNIRSP reduction on CS data in {}'.format(tag(), raw_pol_dir))
                reduce_full_CS(**pol_args)

            print('{}: Using PA&C Modules to fit CU parameters'.format(tag()))
            run_PAC_CUpar(pol_rdx_dir)

        print('{}: Using PA&C Modules to compute demodulation matrices'.format(tag()))
        raw_demod = run_PAC_demod(pol_rdx_dir)

    print('{}: Generating Instrument Polarization Calibration'.format(tag()))
    instpol_calibration = generate_instrument_polarization_calibration(raw_demod, output)

    return instpol_calibration
    


    [docs]def generate_polcal_dark_calibration(SoCCData):

    dark_file_list = SoCCData.get_dark_file_list()
    if len(dark_file_list) != 2:
        raise ValueError('PolCal Calibration Sequence needs exactly 2 dark steps (found {})'.format(len(dark_file_list)))

    output_dark_list = []
    for i in range(2):
        print('{}: Working on dark step #{}'.format(tag(), i+1))
        darkData = Data.FitsData()
        darkData.load_from_filelist(dark_file_list[i])

        print('{}: Averaging data'.format(tag()))
        avgDark = generic.average_FitsData(darkData)

        print('{}: Computing std of data'.format(tag()))
        stdDark = generic.stddev_FitsData(darkData, avgDark)
        avgDark.set_unc(stdDark)
        output_dark_list.append(avgDark)

    print('{}: Computing total average'.format(tag()))
    all_darkData = Data.FitsData()
    all_darkData.load_from_filelist(dark_file_list[0] + dark_file_list[1])
    all_avgData = generic.average_FitsData(all_darkData)
    all_stdData = generic.stddev_FitsData(all_darkData, all_avgData)
    all_avgData.set_unc(all_stdData)

    return output_dark_list[0], output_dark_list[1], all_avgData
    


    [docs]def generate_polcal_gain_calibration(SoCCData, PolDarkCal1, PolDarkCal2, percentile=0.95):
    clear_file_list = SoCCData.get_clear_file_list()
    if len(clear_file_list) != 2:
        raise ValueError('PolCal Calibration Sequence needs exactly 2 clear steps (found {})'.format(len(clear_file_list)))

    dark_list = [PolDarkCal1, PolDarkCal2]
    clear_data_list = []
    for i in range(2):
        print('{}: Working on clear step {}'.format(tag(), i+1))
        clearData = Data.FitsData()
        clearData.load_from_filelist(clear_file_list[i])

        print('{}: Averaging data'.format(tag()))
        avgClear = generic.average_modulated_FitsData(clearData)

        print('{}: Computing std of data'.format(tag()))
        stdDark = generic.stddev_modulated_FitsData(clearData, avgClear)
        avgClear.set_unc(stdDark)

        print('{}: Subtracting dark'.format(tag()))
        generic.dark_correction(avgClear, dark_list[i])
        clear_data_list.append(avgClear)

    print('{}: Computing total average'.format(tag()))
    all_clearData = Data.FitsData()
    all_clearData.load_from_filelist(clear_file_list[0] + clear_file_list[1])
    all_avgData = generic.average_FitsData(all_clearData)
    all_stdData = generic.stddev_FitsData(all_clearData, all_avgData)
    all_avgData.set_unc(all_stdData)

    print('{}: Computing normalization at {} percentile'.format(tag(), percentile*100))
    # Now normalize by the 95th percentile
    cdf = np.sort(np.ravel(all_avgData.data[0]))
    perc95 = int(np.round(cdf.size * percentile))
    norm = cdf[perc95]

    print('{:}: Normalization = {:.4f}. Applying to data'.format(tag(), norm))
    all_avgData.data[0] = all_avgData.data[0] / norm
    all_avgData.unc[0] = all_avgData.unc[0] / norm

    return all_avgData
    


    [docs]def reduce_full_CS(raw_pol_dir, threads=1, dark_cal='PolDarkCal.fits', gain_cal='PolGainCal.fits'):
    """Spawn the parallel reduction of an entire Calibration Sequence.

    This function sets up the right directories and default file names and then hands off the management to
    `manage_CS_rdx`.
    """
    outdir = '{}/proc'.format(raw_pol_dir)
    os.makedirs(outdir, exist_ok=True)

    darkcal_file = '{}/{}'.format(outdir, dark_cal)
    gaincal_file = '{}/{}'.format(outdir, gain_cal)

    print('{}: Scanning polcal files in {}'.format(tag(), raw_pol_dir))
    SoCCData = Data.SoCCData(raw_pol_dir)
    num_steps = SoCCData.num_CS_steps
    print('{}: Found {} calibration sequence steps'.format(tag(), num_steps))

    pol_kwargs = gen_inst_pol_cal_objects(SoCCData, darkcal_file, gaincal_file)

    manage_kwargs = {'output_dir': outdir,
                     'threads': threads}

    manage_CS_rdx(SoCCData, manage_kwargs, pol_kwargs)

    # Create a file that indicates all files were successfully reduced
    with open('{}/rdx_complete'.format(outdir), 'w') as f:
        f.write('CryoNIRSP PolCal reduction of Calibration sequence in {}\nsaved to {}/beam[12] on {}'. \
                format(raw_pol_dir, outdir, time.asctime()))
    


    [docs]def manage_CS_rdx(SoCCData, manage_kwargs, pol_kwargs):
    """Manage the parallel reduction of an entire Calibration Sequence.

    This function constructs a separate job for each CS step and then runs them such that there are never more jobs
    running than the requested number of threads.
    """
    num_steps = SoCCData.num_CS_steps
    numthread = int(manage_kwargs['threads'])
    print('{}: Running CryoNIRSP reduction on all files with {} threads'.format(tag(), numthread))

    # This loop sets up a list of mp processes. They have not been started.
    jobs = []
    pol_kwargs['dark_lock'] = mp.Lock()
    pol_kwargs['gain_lock'] = mp.Lock()
    outpre = manage_kwargs['output_dir']
    for j in range(num_steps):
        jobs.append(mp.Process(target=reduce_single_CS_step,
                               args=[SoCCData, j, outpre],
                               kwargs=pol_kwargs,
                               name='step_{:02}'.format(j)))

    # Now, start jobs and keep starting jobs until the max number of threads is reached
    #  Make a check every second to see if any jobs have finished and thus created room for new ones to start.
    i = 0
    while i < len(jobs):
        while [x.is_alive() for x in jobs].count(True) < numthread:
            jobs[i].start()
            i += 1
            if i >= len(jobs):
                break
        time.sleep(1)

    # Finally, make sure we wait for all jobs to complete before moving on.
    for p in jobs:
        p.join()

    return
    


    [docs]def gen_inst_pol_cal_objects(SoCCData, dark_cal_file, gain_cal_file):

    # Dark
    ######
    dark_prefix = dark_cal_file.split('.fits')[0]
    dark1_filename = '{}_1.fits'.format(dark_prefix)
    dark2_filename = '{}_2.fits'.format(dark_prefix)
    if not (os.path.exists(dark_cal_file) and os.path.exists(dark1_filename) and os.path.exists(dark2_filename)):
        print('{}: Could not find dark cal objects, attempting to generate'.format(tag()))
        DarkCal1, DarkCal2, DarkCalTot = generate_polcal_dark_calibration(SoCCData)
        DarkCal1.write_out(dark1_filename, overwrite=True)
        DarkCal2.write_out(dark2_filename, overwrite=True)
        DarkCalTot.write_out(dark_cal_file, overwrite=True)
    else:
        DarkCal1 = Data.FitsData(dark1_filename)
        print('{}: Loaded {}'.format(tag(), dark1_filename))
        DarkCal2 = Data.FitsData(dark2_filename)
        print('{}: Loaded {}'.format(tag(), dark2_filename))
        DarkCalTot = Data.FitsData(dark_cal_file)
        print('{}: Loaded {}'.format(tag(), dark_cal_file))

    # Gain
    ######
    if not os.path.exists(gain_cal_file):
        print('{}: Could not find {}, attempting to generate'.format(tag(), gain_cal_file))
        GainCal = generate_polcal_gain_calibration(SoCCData, DarkCal1, DarkCal2)
        GainCal.write_out(gain_cal_file, overwrite=True)
    else:
        GainCal = Data.FitsData(gain_cal_file)
        print('{}: Loaded {}'.format(tag(), gain_cal_file))

    pol_kwargs = {'dark_cal': DarkCalTot,
                  'gain_cal': GainCal}

    return pol_kwargs
    


    [docs]def reduce_single_CS_step(SoCCData, CS_step, outdir, dark_cal=None, gain_cal=None,
                          dark_lock=NULLCONTEXT, gain_lock=NULLCONTEXT):

    polData = SoCCData.get_CS_step_data(CS_step)
    print('{}: Loaded CS step {} from {}'.format(tag(), CS_step, SoCCData))

    if dark_cal is not None:
        print('{}: Doing dark correction'.format(tag()))
        with dark_lock:
            print('{}: acquiring multiprocessing dark lock'.format(tag()))
            generic.dark_correction(polData, dark_cal)
            print('{}: releasing multiprocessing dark lock'.format(tag()))

    if gain_cal is not None:
        print('{}: Doing gain correction'.format(tag()))
        with gain_lock:
            print('{}: aqcuiring multiprocessing gain lock'.format(tag()))
            generic.unmodulated_gain_correction(polData, gain_cal)
            print('{}: releasing multiprocessing gain lock'.format(tag()))

    binData = extract_macropixels(polData)

    write_reduced_CS_step(binData, outdir, CS_step)

    return binData
    


    [docs]def extract_macropixels(fitsData):

    nummod = fitsData.headers[0]['CRSP_014']

    hdulist = [pyfits.PrimaryHDU(header=fitsData.headers[0].copy())]
    OGx, OGy = fitsData.data[0].shape
    hdulist[0].header['OGX'] = OGx
    hdulist[0].header['OGY'] = OGy
    for m in range(nummod):
        idx = fitsData.find_data_keyword_idx('CRSP_015', m)[0]
        macro_px = np.median(fitsData.data[idx])[None, None, None]
        ihdu = pyfits.ImageHDU(macro_px, header=fitsData.data_hdus[idx].header.copy())
        numx, numy = fitsData.data[idx].shape
        if numx != OGx or numy != OGy:
            raise ValueError('Modulator states have different data shapes')
        ihdu.header['OGX'] = numx
        ihdu.header['OGY'] = numy
        hdulist.append(ihdu)

    binData = Data.FitsData(allow3D=True)
    binData.hdulist = hdulist
    binData.origin = 'extract macropixels'

    return binData

    [docs]def generate_instrument_polarization_calibration(dmodData, output):

    print('{}: extracting demoulation matrices'.format(tag()))
    numx1, numy1, numl1, nums1, numm1 = dmodData.beam1.data.shape

    if not (numx1 == numy1 == numl1 == 1):
        raise ValueError("This pipeline currently only supports full chip averages")

    if nums1 != 4:
        raise ValueError("The number of Stokes parameters ({}) is not 4. This shouldn't be possible".format(nums1))

    # Right now this is contrived because we force the input shape to be (1, 1, 1) above.
    # Thus we should be able to have the final demod shape be simply (4, M) in order to save space and effort.
    # Instead, let's pretend that we actually need a full frame, just to make sure that harder task is handled correctly
    #  by later machinery.
    OG_shape = (dmodData.beam1.header['OGX'], dmodData.beam1.header['OGY'])
    print('{}: original full-frame shape: {}'.format(tag(), OG_shape))
    big_dmod = np.zeros(OG_shape + (4, numm1), dtype=np.float32)
    big_dmod[:] = dmodData.beam1.data[:, :, 0, :, :]

    final_dmod = Data.DemodulationData()
    final_dmod.init_beam_with_ndarray(big_dmod, 1)
    final_dmod.data_hdus[0].header = dmodData.beam1.header.copy()

    final_dmod.write_out(output, overwrite=True)

    return final_dmod
