    def __repr__(self):

            return 'FitsData created by {}'.format(self.origin)
        else:
            return 'FitsData loaded from {}'.format(self.ref_file)


    [docs]    def load_from_directory(self, directory, searchstr='', max_open=1000, memmap=None):
        """Load a series of FITS files from a directory.

        Each separate file will become a single HDU in this object's data_hdus. A blank PrimaryHDU is provided.

        Parameters
        ----------
        directory : str
            The path to the directory to look in

        searchstr : str
            A string used to narrow down the files loaded. For example, to search for all .fits files starting with
            "image_1", the search string would be "image_1*.fits".

        max_open : int
            The maximum number of open files to keep. Some systems will cause massive errors if too many files are
            open at a time. If loading from a directory with a lot of files, this method will flush() after max_open
            have been loaded.

        memmap : boolean or None
            Use astropy.io.fits' memmap functionality? If None then astropy will make this choice for you.
        """
        searchstr = searchstr or '*{}'.format(CONSTANTS['Data']['suffix'])
        print('{}: Loading files from {}'.format(tag(), directory))
        hdulist = [pyfits.PrimaryHDU()]
        file_list = sorted(glob('{}/{}'.format(directory, searchstr)))

        if len(file_list) == 0:
            raise FileNotFoundError('Did not find any files to load in {}'.format(directory))

        i = 0
        for f in file_list:
            h = pyfits.open(f, memmap=memmap)
            if i > max_open:
                self.hdulist = hdulist
                self.flush()
                i = 0
            print('{}: Loading file {}'.format(tag(), f))
            hdulist.append(h[0])
            i += 1

        hdulist[0].header = hdulist[1].header.copy()
        self.hdulist = pyfits.HDUList(hdulist)
        self.origin = '{}/{}'.format(directory, searchstr)

        """Load data/metadata from a FITS file and make the resulting HDU list accessible to the rest of the class

        Parameters
        ----------
        fitsfile : str
            Path to the FITS file to load. Should probably exist.

        memmap : bool or None
            Whether to use astropy.io.fits' memmap functionality. Set to None to let astropy figure it out for you.
        """
        if not self.is_tmp:
            print('{}: Loading from file {}'.format(tag(), fitsfile))
        self.hdulist = pyfits.open(fitsfile, memmap=memmap)
        self.ref_file = fitsfile

        unc_name = fitsfile.split('.{}'.format(CONSTANTS['Data']['suffix']))[0] \
                   + '_unc.{}'.format(CONSTANTS['Data']['suffix'])
        if os.path.exists(unc_name):
            print('{}: Found uncertainties in {}'.format(tag(), unc_name))
            self.unc = _UncertaintyData(self)
            self.unc.is_tmp = self.is_tmp
            if self.is_tmp:
                self.unc.tmp_file = unc_name
            self.unc.load_from_file(unc_name, memmap=memmap)

        self.origin = fitsfile
    


    [docs]    def init_with_ndarray(self, array, outputfile=None, overwrite=False, autoflush=True):
        """Load data from a numpy.ndarray and construct an astropy.io.fits HDU list

        To provide a uniform data type for all DKIST instruments the follow conventions are enforced:

        1. The PrimaryHDU can (and should) have relevant header information
        2. The PrimaryHDU will have no data
        3. ImageHDU's must have data with 2 or 3 dimensions

        A consequence of point 3 is that an extra, third or fourth (in the case of 2D and 3D data, respectively)
        dimension in the input array will be distributed across multiple ImageHDU's. Input arrays of
        dimensionality > 4 are not currently supported.

        If an output file is not specified then a temporary FITS file will be created to take advantage of
        astropy.io.fits' memmap ability. In this case the FitsData object will be completely lost when it is no
        longer needed. To save the object use the .write_out() method prior to deletion.

        Parameters
        ----------
        array : numpy.ndarray
            Array to be loaded into the FitsData object. Must have 2 or 3 dimensions

        outputfile : str
            If specified, a FITS file will be saved to this path and then reloaded. If None then a temporary file
            will be created.

        overwrite : bool
            If True the outputfile specified will be overwritten.

        autoflush : bool
            If True immediately write to a temp file and reload from disk. This is useful if memory pressure is a
            concern.
        """
        hdulist = [pyfits.PrimaryHDU()]
        hdulist[0].header['FRMARR'] = True
        if len(array.shape) == 3:
            for i in range(array.shape[2]):
                hdulist.append(pyfits.ImageHDU(array[:, :, i]))
        elif len(array.shape) == 4:
            if self._data_list.allow3D:
                for i in range(array.shape[3]):
                    hdulist.append(pyfits.ImageHDU(array[:, :, :, i]))
            else:
                raise TypeError('FitsData will only load 4D arrays if initialized with "allow3D=True"')
        elif len(array.shape) > 3 or len(array.shape) < 2:
            raise TypeError('FitsData currently only supports initialization with ndarrays of shape 2 or 3')
        else:
            #in other words, len(array.shape) == 2
            hdulist.append(pyfits.ImageHDU(array))

        if autoflush:
            # Now write to a file so we don't keep everything in memory
            if not outputfile:
                outputfile = '{}/tmp{}.{}'.format(TMPDIR, ''.join(str(time.time()).split('.')), CONSTANTS['Data']['suffix'])
                self.is_tmp = True
                self.tmp_file = outputfile

            print('{}: initializing array to {}'.format(tag(), outputfile))
            pyfits.HDUList(hdulist).writeto(outputfile, overwrite=overwrite)

            # And then simply load the resulting file
            self.load_from_file(outputfile)
        else:
            self.hdulist = hdulist

        """Flush the current data from memory and reload memmap from the disk
        """
        if self.is_tmp:
            outputfile = self.tmp_file
        else:
            outputfile = '{}/tmp{}.{}'.format(TMPDIR, ''.join(str(time.time()).split('.')), CONSTANTS['Data']['suffix'])

        print('{}: Flushing {} to {}'.format(tag(), self, outputfile))

        self.write_out(outputfile, True)
        for d in self.data_hdus:
            del d.data
        self.is_tmp = True
        self.tmp_file = outputfile
        self.load_from_file(outputfile)
    


    [docs]    def write_out(self, filename, overwrite=True):
        """Save the current object to a FITS file

        Parameters
        ----------
        filename : str
            The location of the resulting FITS file

        overwrite :
            If True, overwrite the filename if it exists. Otherwise complain about a pre-existing file.
        """
        out_hdu_list = pyfits.HDUList()
        loaded_hdus = ['data' in h.__dict__ for h in self.hdulist]
        for i, h in enumerate(self.hdulist):
            h._do_not_scale_image_data = True
            if h.data is None:
                h._do_not_scale_image_data = False
                out_hdu_list.append(h)
                continue
            ih = pyfits.ImageHDU()
            out_hdu_list.append(ih)
            if self.data.allow3D or len(h.data.shape) >= 3:
                data = h.data
            else:
                # Only add the DHS dummy dimension if our data are originally 2D
                data = h.data[None, :, :]
            out_hdu_list[-1].data = data
            out_hdu_list[-1].header = h.header
            if not loaded_hdus[i]:
                del h.data
            h._do_not_scale_image_data = False

        out_hdu_list.writeto(filename, overwrite=overwrite, output_verify='silentfix')

        if self.unc is not None:
            unc_name = filename.split('.{}'.format(CONSTANTS['Data']['suffix']))[0] + \
                       '_unc.{}'.format(CONSTANTS['Data']['suffix'])
            self.unc.write_out(unc_name, overwrite=overwrite)

    def _update_data_hdus(self):
        """Fill data_hdu list with only those HDUs that have data.

        To avoid actually loading the data we look at the required NAXIS header value. It will be 0 (False) for HDUs
        with no data. There are other (maybe better?) ways to do this.
        """
        self.data_hdus = pyfits.HDUList([h for h in self.hdulist if h.header['NAXIS']])

class _DataList():
    """This class exists so that the actual data arrays in FitsData can be accessed without necessarily loading all
    arrays into memory. It also uses np.squeeze to ignore the dummy 3rd dimension in a numpy view sense. Thus, the
    benefits of memmaping are preserved as long as possible."""

    def __init__(self, allow3D=False):
        """Thre's only one property here, a list that will eventually contain pyfits HDU objects"""
        self.data_hdus = pyfits.HDUList()
        self.allow3D = allow3D

    def __del__(self):
        """Keep any loaded arrays out of memory when we're gone"""
        for h in self.data_hdus:
            del h.data

    def __getitem__(self, item):
        """Check that the shape of the actual data is expected and then return the squeezed view of the requested data
        """
        data = self.data_hdus[item].data

        if self.allow3D:
            return data

        if len(data.shape) == 3 and data.shape[0] != 1:
            raise TypeError('Data shape ({}) is wrong'.format(data.shape))

        return np.squeeze(data)

    def __setitem__(self, key, value):
        """Just direct access to underlying array"""
        self.data_hdus[key].data = value

    def __len__(self):
        """Needed so that we can stop using len(FitsData.data_hdus)"""
        return len(self.data_hdus)

class _UncertaintyData(FitsData):

    def __init__(self, parent):

        FitsData.__init__(self)
        self.parent = parent

    def __getitem__(self, item):
        return self.data.__getitem__(item)

    def __setitem__(self, key, value):
        self.data.__setitem__(key, value)

    def __repr__(self):
        return 'Uncertainties associated with {}'.format(self.parent)


    [docs]def concatenate_fitsData(fitsData1, fitsData2):
    """Add in this case means concatenation, NOT a sum of the actual data"""
    if type(fitsData2) is not type(fitsData1):
        raise TypeError('Can only add FitsData derivatives of the same class')

    fitsData1.flush()
    fitsData2.flush()

    out_hdu_list = [fitsData1.hdulist[0]]

    # Go through each HDU separately to reject any HDUs that don't have data
    for h in fitsData1.hdulist:
        if h.header['NAXIS'] != 0: # A good enough proxy for the presence of data
            out_hdu_list.append(h)

    for h in fitsData2.hdulist:
        if h.header['NAXIS'] != 0:
            out_hdu_list.append(h)

    newData = FitsData()
    newData.hdulist = out_hdu_list
    newData.origin = 'sum of {} and {}'.format(fitsData1, fitsData2)

    if fitsData1.unc is not None and fitsData2.unc is not None:
        newUnc = concatenate_fitsData(fitsData1.unc, fitsData2.unc)
        newData.set_unc(newUnc)

    return newData

    [docs]    def read_directory(self, data_dir):
        """Read all files in a directory and populate lists of file names, time stamps, # of ndrs, and current ndr

        Parameters
        ----------
        data_dir : str
            The directory to scan for files
        """
        time_stamps = []
        total_ndrs = []
        current_ndrs = []
        file_list = glob('{}/*{}'.format(data_dir, self.suffix))

        if len(file_list) == 0:
            raise FileNotFoundError('Did not find any valid files with mask {}/*{}'.format(data_dir, self.suffix))

        print('{}: Reading files in {}'.format(tag(), data_dir))
        for f in file_list:
            h = pyfits.open(f)
            time_stamps.append(h[0].header[S122['TimeStamp']])
            total_ndrs.append(h[0].header[S122['NumberOfNDRs']])
            current_ndrs.append(h[0].header[S122['CurrentNDR']])

            h.close()
            del h

        self.file_list = file_list
        self.time_stamps = np.array(time_stamps)
        self.total_ndrs = np.array(total_ndrs)
        self.current_ndrs = np.array(current_ndrs)
    


    [docs]    def discover_ramps(self):
        """Assign each file (ndr) to a specific ramp

        This is done with a dictionary who's keys are the ramp IDs and values are a list of files associated with that
        ramp. This function also ensures that the file lists for each ramp are in correct ndr order.
        """
        unique_times = np.unique(self.time_stamps)

        print('{}: Found {} ramps'.format(tag(), unique_times.size))

        if len(np.unique(self.total_ndrs)) != 1:
            raise ValueError('Not all ramps have the same # NDRs')
        self.numndr = self.total_ndrs[0]
        print('{}: Each ramp has {} NDRs'.format(tag(), self.numndr))

        bad_ramps = {}
        ramp_dict = {}
        for ramp_time in unique_times:
            ndr_idx = np.where(self.time_stamps == ramp_time)[0]

            if ndr_idx.size != self.numndr:
                print('{}: WARNING: Ramp {} is incomplete'.format(tag(), ramp_time))
                target_dir = bad_ramps
            elif np.unique(self.current_ndrs[ndr_idx]).size != self.current_ndrs[ndr_idx].size:
                print('{}: WARNING: Ramp {} has repeated NDR numbers'.format(tag(), ramp_time))
                target_dir = bad_ramps
            else:
                target_dir = ramp_dict

            sort_idx = np.argsort(self.current_ndrs[ndr_idx])

            for i in ndr_idx[sort_idx]:
                try:
                    target_dir[ramp_time].append(self.file_list[i])
                except KeyError:
                    target_dir[ramp_time] = [self.file_list[i]]

        self.ramp_dict = ramp_dict
        self.bad_ramp_dict = bad_ramps

        """Return a RampData object for the specified ramp_id

        Basically we just use the self.ramp_dict database to grab all the files associated with a ramp and pass them
        off to the RampData constructor.

        Parameters
        ----------
        ramp_id : str
            The ID of the ramp requested

        allow_bad : bool
            If True then ramps flagged as "bad" (usually incomplete) can be constructed and returned. Otherwise
            requesting these ramps will result in a ValueError

        Returns
        -------
        `Data.RampData`
            Object containing the ndrs of the requested ramp

        """
        if ramp_id in self.bad_ramp_dict.keys():
            if allow_bad:
                target_dict = self.bad_ramp_dict
            else:
                raise ValueError('Ramp {} is flagged as bad. Use allow_bad=True to force load.'.format(ramp_id))
        elif ramp_id not in self.ramp_dict.keys():
            raise ValueError('Ramp {} not found'.format(ramp_id))
        else:
            target_dict = self.ramp_dict

        RD = RampData(target_dict[ramp_id])

        return RD

    [docs]    def load_from_filelist(self, file_list):
        """Open and load each file in the given file list

        There is also a simple check to make sure the file list is in ndr/read order.

        All of the data arrays are loaded into a single (N, x, y) numpy array and the corresponding headers are in a
        single, length N, array.

        Parameters
        ----------
        file_list : list
            A list of all files (ndr's) associated with the ramp. These must be in read-order.
        """
        datalist = []
        headerlist = []

        ndrcheck = -np.inf
        for f in file_list:
            h = pyfits.open(f)

            test_ndr = h[0].header[S122['CurrentNDR']]
            if test_ndr <= ndrcheck:
                raise ValueError('NDR order appears to be wrong')

            datalist.append(h[0].data)
            headerlist.append(h[0].header)
            ndrcheck = test_ndr

        self.data = np.stack(datalist)
        self.data_shape = self.data.shape[1:]
        self.headers = headerlist
        self.ndr = test_ndr
    


    [docs]    def correct_camera_specifics(self):
        """Identify the camera and mode used to collect the data and perform specific pre-processing steps, if needed.

        The camera is used to determine the gain.

        Fast mode data have the first ndr removed.

        Slow mode data are inverted.
        """
        instrument = self.headers[0][S122['CryoInstrument']]
        if instrument == 1:
            self.instrument = 'SP'
        elif instrument == 2:
            self.instrument = 'CI'
        else:
            raise ValueError('Unable to determine instrument ({})'.format(instrument))

        self.gain = CONSTANTS['Data']['{}_CAMERA_GAIN'.format(self.instrument)]
        print('{}: Loaded gain of {:.1f} for {} instrument'.format(tag(), self.gain, self.instrument))

        self.camera_mode = self.headers[0][S122['CameraMode']]
        print('{}: Applying corrections for mode {}'.format(tag(), self.camera_mode))
        if self.camera_mode == 'SFRAME':
            print('{}: Inverting data'.format(tag()))
            self.invert_16bit_data()

        if self.camera_mode == 'FFRAME':
            print('{}: Removing first ndr'.format(tag()))
            self.data = self.data[1:, :, :]
            self.ndr -= 1
    


    [docs]    def construct_time_vector(self):
        """Read file headers and construct a vector containing the time stamp of each ndr in the ramp
        """
        exp_time = self.headers[0][S122['CryoExpTime']]
        dt = self.headers[0][S122['CryoDT']]
        frame_time = (exp_time - (self.ndr - 1) * dt) / (self.ndr - 1)
        print('{}: FRAME time: {:.2f}. DT: {:.2f}'.format(tag(), frame_time, dt))
        self.time_vec = (np.arange(self.ndr) * (frame_time + dt)) / 1000.0 # time in seconds
    


    [docs]    def check_dead_beef(self):
        """Check for scrambled data by identifying the location of two signals purposely inserted into the data
        """
        # Is self.x_size needed? Why not just 0::self.channel_width?
        test = self.data[:, 0, 0:self.data_shape[0]:self.channel_width]

        dead = test == 57005
        beef = test == 48879

    [docs]class CSStep():
    """This class allows for an easy way to quickly compare SPEC-0122 style headers to determine if they come from the
    same Calibration Sequence (CS) step. Each step in a CS is defined by the configuration of the GOS, namely the
    status of the polarizer and retarder (in or out) and the angle of each. Because some CS schemes call for some
    GOS configurations to repeat a check is also made against the observation time for each object; the defaults for
    the ViSP Pipeline define a maximum difference in time where two exposures are considered different CS steps
    regardless of GOS configuration.

    This class can also be sorted. In this case, only the observation time is taken into account.

    Finally, this class is hashable for use in dictionaries. The hash is only based on the GOS configuration so that
    all objects from the same CS step result in the same hash. Python also checks that dictionary keys are equal so any
    conflicts with repeated CS steps are avoided.
    """

    def __init__(self, header):
        """Initialize and read the GOS configuration and time from a SPEC-0122 FITS header.

        Parameters
        ----------
        header : `astropy.io.fits.header.Header`
            Object containing a SPEC-0122 compliant header

        """
        self.pol_in = header[S122['GOSPolarizer']]
        self.pol_theta = header[S122['PolarizerAngle']]
        self.ret_in = header[S122['GOSRetarder']]
        self.ret_theta = header[S122['RetarderAngle']]
        self.dark_in = header[S122['GOSLevel0']] == 'Dark'
        self.obs_time = Time(header['DATE-OBS'])

    def __repr__(self):
        return 'CC with Pol = {}:{}, Ret = {}:{}, Dark = {}. Taken at {}'.format(self.pol_in,
                                                                                 self.pol_theta,
                                                                                 self.ret_in,
                                                                                 self.ret_theta,
                                                                                 self.dark_in,
                                                                                 self.obs_time.fits)

    def __eq__(self, other):
        """Two steps are equal if they have the same GOS configuration and are taken within some package-defined time
        of each other."""
        if not isinstance(other, CSStep):
            return False

        for item in ['pol_in', 'pol_theta', 'ret_in', 'ret_theta', 'dark_in']:
            if getattr(self, item) != getattr(other, item):
                return False

        tdelt = abs(self.obs_time - other.obs_time)
        if tdelt.sec > float(CONSTANTS['Data']['max_CS_step_time_sec']):
            return False

        return True

    def __lt__(self, other):
        """Only based on time"""
        return self.obs_time < other.obs_time

    def __hash__(self):
        """Only based on GSO configuration so that all objects from the same CS step hash the same"""
        return hash((self.pol_in, self.pol_theta, self.ret_in, self.ret_theta))

class _TaskData(FitsData):
    """_TaskData is a class template that is used to define specific *Data classes that are designed to only hold data
    from a specific Instrument Program Task (DKIST004). It inherits all of the functionality of `FitsData` and defines
    an extra method that loads data from a directory, provided it matches the class's task. After loading, this class
    behaves exactly like `FitsData`.
    """

    def __repr__(self):
        repstr = '{} loaded from {}'.format(self.__class__.__name__, self.ref_dir)
        return repstr

    def load_from_directory(self, data_dir, suffix='', maxopen=1000):
        """Load DHS-style data from a directory.

        It is assumed that all relevant files in the directory have SPEC-0122 headers and a single HDU. In other words,
        each file is exactly one SysOutput. Only files whose InstrumentProgramTask (DKIST004) is equal to this class's
        task attribute will be loaded.

        Parameters
        ----------
        data_dir : str
            Path to the directory containing data to load

        suffix : str
            Optional file mask. The full mask is DATA_DIR/*SUFFIX. Defaults to ".fits"
        """
        suffix = suffix or CONSTANTS['Data']['suffix']
        file_list = sorted(glob('{}/*{}'.format(data_dir, suffix)))
        hdulist = []

        i = 0
        for f in file_list:
            h = pyfits.open(f)
            if all([str(h[0].header[S122[k]]) in str(self.task_dict[k]) for k in self.task_dict.keys()]):
                if i > maxopen:
                    self.hdulist = hdulist
                    self.flush()
                    i = 0

                hdulist.append(h[0])
                self.file_list.append(f)
                i += 1

        if len(hdulist) > 0:
            hdulist = [pyfits.PrimaryHDU(data=None, header=hdulist[0].header)] + hdulist
            self.hdulist = pyfits.HDUList(hdulist)
            self.ref_dir = data_dir
        else:
            raise FileNotFoundError('{}: Did not find any {} images with {}/*{}'.format(tag(), self.__class__.__name__,
                                                                                        data_dir, suffix))
