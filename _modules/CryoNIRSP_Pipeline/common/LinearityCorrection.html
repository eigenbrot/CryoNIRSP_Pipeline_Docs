    [docs]def main(target_dir, output_dir, threshold_file, bad_px_file, # LUT_file,
         x0=None, y0=None, x_size=None, y_size=None, order=2, overwrite=False):
    """Use a collection of raw DHS non-destructive reads (NDRs) to produce linearity corrected data.

    This is essentially a pre-processing step for all data before they can be ingested properly by the rest of the
    pipeline.

    Parameters
    ----------
    target_dir : str
        Path to directory containing NDRs for any number of separate ramps

    output_dir : str
        Path to save the linearity corrected ramp data

    threshold_file : str
        Path to a numpy save (.npy) file that contains the linearity threshold for each pixel in the subarray to use
        (see below).

    bad_px_file : str
        Path to a numpy save (.npy) file that contains a bad pixel mask for each pixel in the subarray to use (see below)

    x0 : int
        The origin (in pixel coordinates) of first dimension for the subarry to use

    y0 : int
        The origin (in pixel coordinates) of second dimension for the subarry to use


    x_size : int
        The size, in pixels, of the subarry in the first dimension

    y_size : int
        The size, in pixels, of the subarray in the second dimension

    order : int
        The polynomial order to use when fitting the linear portion of each ramp

    overwrite : bool
        If True then any already-corrected output files will be overwritten
    """

    if order < 1:
        raise ValueError('Order cannot be less than 1. (You asked for {})'.format(order))

    os.makedirs(output_dir, exist_ok=True)

    if threshold_file is not None:
        thresh = np.load(threshold_file)
    else:
        thresh = None
    bad_px = np.load(bad_px_file)
    # LUT = np.load(LUT_file)

    allRamps = Data.RampAggregator(target_dir)

    for ramp_id in allRamps.ramp_dict.keys():

        print('{}: Correcting ramp {}'.format(tag(), ramp_id))
        rampData = allRamps.get_ramp_data(ramp_id)
        rampData.init_subarray(x0, y0, x_size, y_size)

        # new_data = correct_single_ramp_LUT(rampData, LUT, bad_px, threshold_vec=thresh)[None,:,:]
        new_data = correct_single_ramp(rampData, bad_px, order=order, threshold_vec=thresh)[None,:,:]

        date_id = Time(ramp_id).datetime.strftime("%Y%m%dT%H%M%S.%f")
        outname = '{}/CRSP_{}_{}_{}.{}'.format(output_dir, rampData.instrument, rampData.camera_mode, date_id,
                                               CONSTANTS['Data']['suffix'])

        pyfits.PrimaryHDU(new_data, header=rampData.headers[0]).writeto(outname, overwrite=overwrite)
    



    [docs]def correct_single_ramp(rampData, bad_px_vec, order=2, threshold_vec=None, plot_px=None):
    """Perform linearity correction on a single ramp. This is the heart of this module.

    The correction pipeline has the following steps:

    1. Use ramp data to conform the threshold values to the current data

    2. Identify pixels above the threshold and mask them out

    3. Fit polynommials  region of each pixel

    4. Use the polynomial fit coefficients to compute the "true" pixel value at the end of the ramp

    Parameters
    ----------
    rampData : `Data.RampData`
        Object containing the ramp to be corrected. This object must have had a subarray initialized.

    threshold_vec : `numpy.ndarray`
        Vector containing the upper threshold of the linear region for each pixel. Must have the same shape as
        rampData.flattened_subarray.shape[1:]

    bad_px_vec : `numpy.ndarray`
        Vector containing as bad pixel mask. Must have shape equal to rampData.flattened_subarray.shape[1:]

    order : int
        Order of polynomial to fit to the linear regions of each pixel

    Returns
    -------
    `numpy.ndarray`
        Corrected pixel values for all pixels in the rampData subarray. Same shape as rampData.subarray
    """
    if threshold_vec is not None:
        bias = rampData.flattened_subarray[0]
        print('{}: Using pre-computed threshold values'.format(tag()))
        print('{:}: Correcting threshold with gain {:.1f} and average bias {:.1f}'.format(tag(),
                                                                                          rampData.gain,
                                                                                          np.mean(bias)))
        corrected_threshold = threshold_vec / rampData.gain + bias
    else:
        print('{}: Computing threshold values'.format(tag()))
        corrected_threshold = find_threshold(rampData.flattened_subarray)

    print('{}: Computing masks'.format(tag()))
    threshold_mask, first_nan = get_threshold_mask(rampData.flattened_subarray, corrected_threshold)

    print('{}: Checking masks'.format(tag()))
    fit_column_mask = check_mask_vs_fitorder(first_nan, order)

    print('{}: Computing fit coefficients'.format(tag()))
    # For now, we only fit with the order requested (fit_column_mask[0]). In the future we might need to add
    # the ability to fit for lower orders as well (fit_column_mask[1:-1]).
    threshold_mask[:, fit_column_mask[0]] = True
    coeffs = masked_polyfit(rampData.time_vec, rampData.flattened_subarray, order, threshold_mask, first_nan)

    if plot_px is not None:
        make_debug_plots(rampData, corrected_threshold, first_nan, coeffs, plot_px)
        input('Waiting...')

    print('{}: Calculating corrected data values'.format(tag()))
    # We take coeffs[-2:] because we only want the linear part
    #  Also, is it dumb to use polyval(coeffs[-2:], ...) and then subtract coeffs[-1]? Probably.
    tmp = rampData.gain * (np.polyval(coeffs[-2:], rampData.time_vec[-1]) - coeffs[-1])
    tmp[bad_px_vec] = np.nan
    output = np.float32(np.reshape(tmp, rampData.subarray_shape))

    return output
    


    [docs]def correct_single_ramp_LUT(rampData, LUT, bad_px_vec, threshold_vec=None):
    """Perform linearity correction on a single ramp. This is the heart of this module.

    The correction pipeline has the following steps:

    1. Use ramp data to conform the threshold values to the current data

    2. Identify the last NDR prior to saturation

    3. For each pixel, apply the LUT polynomials to the NDRs found in 2

    4. Convert to flux by dividing by the time of the NDRs found in 2

    Parameters
    ----------
    rampData : `Data.RampData`
        Object containing the ramp to be corrected. This object must have had a subarray initialized.

    LUT : `numpy.ndarray`
        Array of shape (order + 1, x*y) containing lookup-table polynomial coefficients for each pixel

    threshold_vec : `numpy.ndarray`
        Vector containing the upper threshold of the linear region for each pixel. Must have the same shape as
        rampData.flattened_subarray.shape[1:]

    bad_px_vec : `numpy.ndarray`
        Vector containing as bad pixel mask. Must have shape equal to rampData.flattened_subarray.shape[1:]

    Returns
    -------
    `numpy.ndarray`
        Corrected pixel values for all pixels in the rampData subarray. Same shape as rampData.subarray
    """
    bias = rampData.flattened_subarray[0]
    if threshold_vec is not None:
        print('{}: Using pre-computed threshold values'.format(tag()))
        print('{:}: Correcting threshold with gain {:.1f} and average bias {:.1f}'.format(tag(),
                                                                                          rampData.gain,
                                                                                          np.mean(bias)))
        corrected_threshold = threshold_vec / rampData.gain + bias
    else:
        print('{}: Computing threshold values'.format(tag()))
        corrected_threshold = find_threshold(rampData.flattened_subarray)

    print('{}: Finding highest non-saturated NDRs'.format(tag()))
    threshold_mask, first_nan = get_threshold_mask(rampData.flattened_subarray, corrected_threshold)

    last_non_nan = first_nan - 1
    only_saturated_idx = np.where(last_non_nan <= 0)[0]
    last_NDR_vals = rampData.flattened_subarray[last_non_nan, np.arange(np.prod(rampData.subarray_shape))] - bias

    print('{}: Applying LUT values'.format(tag()))
    tmp = rampData.gain * last_NDR_vals * np.polyval(LUT, last_NDR_vals)
    flux = tmp / rampData.time_vec[last_non_nan]
    flux[bad_px_vec] = np.nan
    flux[only_saturated_idx] = np.nan
    output = np.float32(np.reshape(flux, rampData.subarray_shape))

    return output
    



    [docs]def find_threshold(flat_data, linear_ddx_threshold=100., threshold_knee=0.9):
    """Find the threshold value at which ramp data departs from linearity

    The threshold is computed separately for all pixels in the image and is found via the second derivative of electrons
    per NDR. A check is made for pixels that are linear over all NDRs provided.

    Parameters
    ----------
    flat_data : `numpy.ndarray`
        Array with shape (N, x*y) where N is the number of NDRs and (x, y) are the detector image dimensions

    linear_ddx_threshold : float
        If the second derivative of the ramp data is below this level then the ramp is assumed to be linear across all
        NDRs and thus the threshold will be the value of the last NDR

    threshold_knee : float
        A scale factor to apply to the location of maximum second derivative so that the reported threshold is slightly
        below the saturation limit. Should REALLY be less than 1.

    Returns
    -------
    `numpy.ndarray`
        Array with shape (x*y). The upper linearity threshold for each pixel.
    """

    ddx = np.gradient(np.gradient(flat_data, axis=0), axis=0)

    # If the ramp is basically linear across all NDRs then ddx will be mostly constant. Check for that here.
    all_linear = np.where(np.std(ddx, axis=0) < linear_ddx_threshold)

    max_delta_idx = np.argmin(ddx, axis=0)
    max_delta_idx[all_linear] = flat_data.shape[0] - 1 # If the ramp is linear across all NDRs then set the index of last linear NDR to the final NDR

    # move the threshold a "bit" below the value of max change. Highly scientific.
    threshold = flat_data[max_delta_idx, range(flat_data.shape[1])] * threshold_knee

    return threshold
    


    [docs]def get_threshold_mask(data, threshold):
    # From Andre's cnToolbox.py
    """Compute a mask that masks out pixels above the given threshold

    Also note at which NDR each pixel crosses the threshold.

    Andre's doc:
    get a mask with 1 after first occurance of threshold in column
    get first_nan vector with index of first above threshold
    both are ndarrays

    Parameters
    ----------
    data : `numpy.ndarray`
        Array with shape (N, x*y) where N is the number of NDRs and (x, y) are the ramp subarray dimensions
        
    threshold : `numpy.ndarray`
        Array of shape (x*y) containing the upper limit of the linear region for each pixel


    Returns
    -------
    `numpy.ndarray`
        Boolean array with the same shape as ``data``. True corresponds to readings above the threshold value

    `numpy.ndarray`
        Vector of shape (x*y) containing the NDR # at which each pixel goes above the threshold value
    """
    if threshold is None:
        return np.zeros(data.shape), np.ones(data.shape[1])*data.shape[0]

    max_ind = data.shape[0] - 1
    # find all points with values higher than threshold
    b = ma.masked_where(data >= threshold, data)

    # bugfix 3 Jul 2019: if condition in ma.masked_where is True nowhere
    # the mask will be a single boolean False rather than an array of False
    if np.size(b.mask) == 1:
        # None of the data is bad so make make mask an array of FALSEs
        b.mask = np.zeros(data.shape[1])

    # argmax returns the *first* instance
    first = np.argmax(b.mask, axis=0)

    # argmax does not work if all 1 or all 0;
    # all returns True if all 1s
    tester = np.all(~b.mask, axis=0)
    first[tester] = max_ind

    ind = np.indices(data.shape)[0]
    mask = (ind >= first)

    return mask, first
    


    [docs]def check_mask_vs_fitorder(firstNan, order):
    # From Andre's cnToolbox.py
    """Check which columns can be fitted with what order of polynomial

    Any pixels that do not have enough "good" ndrs to fit the given order are tested against all lower orders. Thus,
    each pixel is assigned its maximum possible fit order.

    Parameters
    ----------
    firstNan : `numpy.ndarray`
        1D array containing the first ndr at which each pixel becomes non-linear

    order : int
        Maximum fitting order that will be used to fit the linear regions

    Returns
    -------
    `numpy.ndarray`
        Binary array of shape (order + 1, x*y) that contains the BAD pixel mask for each order. In other words,
        mask[i, :] == 0 are all the pixels in order i that CAN be fit with that order. The first dimension indexes
        order in reverse, so mask[0] is the highest fit order and mask[-1] are pixesl that cannot be fit with any order.
    """

    # needs to be list as they could have different lengths
    fit_column_index = []
    # 3D mask that contains mask for each highest order possible
    mask = np.ones((order+1, firstNan.shape[0]), dtype=np.int8)
    # make firstNan masked array so we can disregard used ones
    firstNan = ma.array(firstNan, mask=np.zeros(firstNan.shape))
    # are there any series where number of data points is
    # not sufficient for order of polynom?
    for i in range(order, 0, -1):
        # can do full order
        tmp = np.int32(ma.where(firstNan > i)[0])
        fit_column_index.append(tmp)
        # mask the used values
        firstNan.mask[tmp] = 1
        # for the overall mask we want 0 for the values we can use
        # for this order
        mask[order-i, tmp] = 0

    # remaining columns have one or less data points
    tmp = np.int32(ma.where(firstNan <= i)[0])
    fit_column_index.append(tmp)

    # for the overall mask we want 0 for the values we can use for that order
    mask[-1, tmp] = 0

    for i in range(order):
        print('{}: {} pixels will be fit with order = {}'.format(tag(), len(fit_column_index[i]), order - i))
    print('{}: {} pixels cannot be fit'.format(tag(), len(fit_column_index[-1])))

    return mask
    


    [docs]def masked_polyfit(time, data, order, mask, first_nan, fit=False):
    """Fit each pixel with a polynomial

    Andre's doc:
    using standar masked polyfit
    !!! ma.polyfit mask up to the shortest data series when 2D masked y is given
    i.e. truncates longer series -> solve with loop

    Parameters
    ----------
    time : `numpy.ndarray`
        The time (in seconds) of each NDR

    data : `numpy.ndarray`
        Array of shape (N, x*y) where N is the number of NDRs and (x, y) is the ramp's subarray shape

    order : int
        Order of polynomial to use for the fit

    mask : `numpy.ndarray`
        Boolean array of shape (N, x*y) corresponding to bad pixel mask for each order

    first_nan : `numpy.ndarray`
        Vector of shape (x*y) containing the NDR # at which each pixel goes above the linear threshold

    fit : bool
        If True return the best fit along with the coefficients

    Returns
    -------
    `numpy.ma.core.MaskedArray`
        Array of shape (order + 1, x*y) containing the best-fit polynomial coefficients for each pixel
    """
    maskedY = ma.masked_array(data, mask=mask)
    # full = np.where(first_nan > order)[0]

     # initialize coef array with NaNs
    coef = np.zeros((order + 1, data.shape[1])) + np.nan

    # how many unique lengths of data series
    myl = np.unique(first_nan)
    # only use the ones that can fit the chosen order
    myl = myl[myl > order]

    if myl.size > 0:
        # have to loop through all lengths otherwise collapsed mask causes issues
        for i in myl:
            wo = np.where(first_nan == i)[0]
            # print('data series length', i)
            # print('percentage of points', 100*len(wo)/(data.shape[1]))
            coef[:, wo] = ma.polyfit(time, maskedY[:, wo], order)

        if fit:
            fitX = np.tile(time, (data.shape[1], 1))
            fit = np.polyval(coef, fitX.T)
            fit = ma.masked_array(fit, mask=mask)
            return coef, fit
        return coef
    raise ValueError('no valid data points for fit')
    


    [docs]def make_debug_plots(rampData, threshold, first_NaN, coeffs, pixels_to_plot):

    import matplotlib.pyplot as plt
    from matplotlib.backends.backend_pdf import PdfPages as PDF
    pp = PDF('linearity_plots.pdf')

    for pidx in pixels_to_plot:
        idx = np.ravel_multi_index(pidx, rampData.subarray_shape)
        fig = plt.figure(figsize=(12, 8))
        ax = fig.add_subplot(111)
        ax.set_xlabel('NDR #')
        ax.set_ylabel('e-')
        ax.plot(rampData.time_vec, rampData.flattened_subarray[:, idx], 'k-')
        ax.axvline(rampData.time_vec[first_NaN[idx]], ls='--', color='k', alpha=0.5)
        ax.axhline(threshold[idx], ls='--', color='k', alpha=0.5)
        fit_values = np.polyval(coeffs[:, idx], rampData.time_vec)
        ax.plot(rampData.time_vec, fit_values, ls=':', color='r')
        ax.scatter(rampData.time_vec[-1], fit_values[-1], marker='o', color='g', s=20)
        ax.text(rampData.time_vec[-1], fit_values[-1], '{:.2f} '.format(rampData.gain * (fit_values[-1] - coeffs[-1, idx])),
                horizontalalignment='right', verticalalignment='center', fontsize=10)
        ax.set_title('Pixel {}. (ds9 {})'.format(pidx, np.array(pidx[::-1]) + 1))

        dax = ax.twinx()
        dax.set_ylabel(r'$\frac{d^2e-}{dx^2}$')
        dax.spines['right'].set_color('b')
        dax.yaxis.label.set_color('b')
        dax.tick_params(axis='y', colors='b')
        ddx = np.gradient(np.gradient(rampData.flattened_subarray[:, idx]))
        dax.plot(rampData.time_vec, ddx, 'b-', alpha=0.4)
        dax.text(0.8, 0.1, '{:.1f}'.format(np.std(ddx)), transform=dax.transAxes, color='b', fontsize=8)

        pp.savefig(fig)

    pp.close()
