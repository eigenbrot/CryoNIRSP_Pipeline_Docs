    import os
import time
import numpy as np
from CryoNIRSP_Pipeline.common import generic, Data, LinearityCorrection
from CryoNIRSP_Pipeline.common.tag import tag
from CryoNIRSP_Pipeline.data import CONSTANTS, S122


    [docs]def main(dark_dir, instrument, outpre, linearize=True):

    if linearize:
        lin_dir = os.path.join(dark_dir, 'linearized')
        LinearityCorrection.main(dark_dir, lin_dir, **CONSTANTS['LinearityKwargs'])
        dark_dir = lin_dir

    if instrument == 'SP':
        darkData = Data.SPDarkData(dark_dir)
    elif instrument == 'CI':
        darkData = Data.CIDarkData(dark_dir)
    else:
        raise ValueError("'instrument' needs to be either 'SP' or 'CI'")

    validate_dark_task_data(darkData)

    print('{}: Averaging data'.format(tag()))
    avgDark = generic.average_FitsData(darkData)

    print('{}: Computing std of data'.format(tag()))
    stdDark = generic.stddev_FitsData(darkData, avgDark)
    avgDark.set_unc(stdDark)

    validate_dark_calibration(avgDark)

    print('{}: Generating Dark Calibration'.format(tag()))
    dark_calibration = generate_dark_calibration(avgDark, dark_dir, outpre)

    return dark_calibration
    


    [docs]def validate_dark_task_data(darkData):

    IP0 = darkData.headers[1][S122['InstrumentProgramID']]
    for h in darkData.data_hdus:
        if h.header[S122['InstrumentProgramID']] != IP0:
            raise ValueError('Not all frames have the same IP ID')
    


    [docs]def validate_dark_calibration(avgDark):

    if np.min(avgDark.data[0]) < 0:
        raise ValueError('Dark Calibration has negative values')

    std_frac = np.mean(avgDark.data[0] / avgDark.unc[0])
    if std_frac < 1:
        raise ValueError('Dark Calibration stddev is too high ({:.3f} of mean)'.format(std_frac))
    


    [docs]def generate_dark_calibration(avgData, dark_dir, outpre):

    outname = outpre + '.' + CONSTANTS['Data']['suffix']
    avgData.origin = 'Dark Calibration'
    avgData.headers[0]['HISTORY'] = 'Generated with data from "{}" on {}'.format(os.path.abspath(dark_dir),
                                                                                 time.asctime())
    avgData.write_out(outname, overwrite=True)

    return avgData
