    [docs]def average_FitsData(fitsData):
    """Average ALL hdus into a single frame

    The DATE header keywords are updated to be the average of all used frames

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Data to be averaged

    Returns
    -------
    `Data.FitsData`
        Will have a single data_hdu that contains the average of all input data_hdus
    """
    numx, numy = fitsData.data[0].shape
    hdulist = [pyfits.PrimaryHDU(header=fitsData.headers[0])]

    print('{}: Averaging all HDUs from {}'.format(tag(), fitsData))
    output = np.zeros((numx, numy), dtype=np.float32)
    if fitsData.unc is not None:
        out_unc = np.zeros(output.shape, dtype=output.dtype)
    start_times = []
    end_times = []
    for i, h in enumerate(fitsData.data_hdus):
        output += fitsData.data[i]
        if fitsData.unc is not None:
            out_unc += fitsData.unc[i]**2
        start_times.append(Time(h.header['DATE-BGN']))
        end_times.append(Time(h.header['DATE-END']))
    hdulist.append(pyfits.ImageHDU(output / len(fitsData.data_hdus), fitsData.headers[1]))

    new_start_time = min(start_times)
    new_end_time = max(end_times)
    for h in hdulist:
        h.header['DATE-BGN'] = new_start_time.fits
        h.header['DATE-END'] = new_end_time.fits
        h.header['DATE-OBS'] = (new_start_time + (new_end_time - new_start_time) / 2).fits
    newData = Data.FitsData()
    newData.hdulist = hdulist
    newData.origin = 'average_FitsData'
    if fitsData.unc is not None:
        newData.set_unc(np.sqrt(out_unc) / len(fitsData.data_hdus))

    return newData
    


    [docs]def stddev_FitsData(fitsData, average=None):
    """Compute the per-pixel standard deviation across ALL HDUs present in the input data

    The DATE header keywords are updated to be the average of all data frames used

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Data to compute the stddev of

    average : `Data.FitsData`
        A precomputed total average of the input data. Almost certainly the output of `average_FitsData`. If no average
        is provided then `average_FitsData` is used to compute it.

    Returns
    -------
    `Data.FitsData`
        The standard deviation across all frames in the input data. This object will have a single data_hdu.
    """
    if average is None:
        average = average_FitsData(fitsData)

    numx, numy = fitsData.data[0].shape
    hdulist = [pyfits.PrimaryHDU(header=fitsData.headers[0])]
    avg_data = average.data[0]

    print('{}: Computing sigma on all HDUs from {}'.format(tag(), fitsData))
    output = np.zeros((numx, numy), dtype=np.float32)
    start_times = []
    end_times = []
    for i, h in enumerate(fitsData.data_hdus):
        output += (fitsData.data[i] - avg_data)**2
        start_times.append(Time(h.header['DATE-BGN']))
        end_times.append(Time(h.header['DATE-END']))
    hdulist.append(pyfits.ImageHDU(np.sqrt(output / len(fitsData.data_hdus)), fitsData.headers[1]))

    new_start_time = min(start_times)
    new_end_time = max(end_times)
    for h in hdulist:
        h.header['DATE-BGN'] = new_start_time.fits
        h.header['DATE-END'] = new_end_time.fits
        h.header['DATE-OBS'] = (new_start_time + (new_end_time - new_start_time) / 2).fits
    newData = Data.FitsData()
    newData.hdulist = hdulist
    newData.origin = 'stddev_FitsData'

    return newData
    


    [docs]def average_modulated_FitsData(fitsData):
    """Average all data from the same modulator state together

    The DATE header keywords are updated to be the average of all the frames corresponding to a particular modulator
    state

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Data to be averaged together. Different modulator states are identified via their SPEC-0122 headers.

    Returns
    -------
    `Data.FitsData`
        The average of each modulator state will live in a separate data_hdu
    """
    numx, numy = fitsData.data[0].shape
    hdulist = [pyfits.PrimaryHDU(header=fitsData.headers[0])]
    nummod = fitsData.headers[0][S122['NumberOfModulatorStates']]

    if fitsData.unc is not None:
        unc_output = np.zeros((numx, numy, nummod), dtype=np.float32)

    for i in range(nummod):
        pos_idx = fitsData.find_data_keyword_idx(S122['CurrentState'], i)
        if len(pos_idx) > 0:
            output = np.zeros((numx, numy), dtype=np.float32)
            start_times = []
            end_times = []
            print('{}: Averaging modulator state {}'.format(tag(), i))
            for j in pos_idx:
                print('{}:  using HDU {}'.format(tag(), j))
                output += fitsData.data[j]
                if fitsData.unc is not None:
                    unc_output[:, :, i] += fitsData.unc[j]**2
                start_times.append(Time(fitsData.data_hdus[j].header['DATE-BGN']))
                end_times.append(Time(fitsData.data_hdus[j].header['DATE-END']))
            hdulist.append(pyfits.ImageHDU(output / len(pos_idx), header=fitsData.data_hdus[pos_idx[0]].header))
            if fitsData.unc is not None:
                unc_output[:, :, i] = np.sqrt(unc_output[:, :, i]) / len(pos_idx)
            new_start_time = min(start_times)
            new_end_time = max(end_times)
            hdulist[-1].header['DATE-BGN'] = new_start_time.fits
            hdulist[-1].header['DATE-END'] = new_end_time.fits
            hdulist[-1].header['DATE-OBS'] = (new_start_time + (new_end_time - new_start_time) / 2).fits

    newData = Data.FitsData()
    newData.hdulist = hdulist
    newData.origin = 'average_modulated_FitsData'
    if fitsData.unc is not None:
        newData.set_unc(unc_output)

    return newData
    


    [docs]def stddev_modulated_FitsData(fitsData, average=None):
    """Compute the per-pixel standard deviation for each modulator state present in the input data

    The DATE header keywords are updated to be the average of the data frames associated with each modulator state.

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Data to compute the stddev of

    average : `Data.FitsData`
        A precomputed total average of the input data. This object must have the same number of modulator states as the
        input data. Almost certainly the output of `average_modulated_FitsData`. If no average is provided then
        `average_modulated_FitsData` is used to compute it.

    Returns
    -------
    `Data.FitsData`
        The standard deviation for each modulator state in the input data. This object will have a single data_hdu for
        each modulator state.
    """

    if average is None:
        average = average_modulated_FitsData(fitsData)

    numx, numy = fitsData.data[0].shape
    hdulist = [pyfits.PrimaryHDU(header=fitsData.headers[0])]
    nummod = fitsData.headers[0][S122['NumberOfModulatorStates']]

    for i in range(nummod):
        pos_idx = fitsData.find_data_keyword_idx(S122['CurrentState'], i)
        if len(pos_idx) > 0:
            output = np.zeros((numx, numy), dtype=np.float32)
            start_times = []
            end_times = []
            avg_idx = average.find_data_keyword_idx(S122['CurrentState'],i)
            if len(avg_idx) == 0:
                raise ValueError('Could not find average value for modulator {}'.format(i))
            avg_data = average.data[avg_idx[0]]
            print('{}: Computing stddev for modulator state {}'.format(tag(), i))
            for j in pos_idx:
                print('{}:  using HDU {}'.format(tag(), j))
                output += (fitsData.data[j] - avg_data)**2
                start_times.append(Time(fitsData.data_hdus[j].header['DATE-BGN']))
                end_times.append(Time(fitsData.data_hdus[j].header['DATE-END']))

            hdulist.append(pyfits.ImageHDU(np.sqrt(output / len(pos_idx)),
                                           header=fitsData.data_hdus[pos_idx[0]].header))
            new_start_time = min(start_times)
            new_end_time = max(end_times)
            hdulist[-1].header['DATE-BGN'] = new_start_time.fits
            hdulist[-1].header['DATE-END'] = new_end_time.fits
            hdulist[-1].header['DATE-OBS'] = (new_start_time + (new_end_time - new_start_time) / 2).fits

    newData = Data.FitsData()
    newData.hdulist = hdulist
    newData.origin = 'stddev_modulated_FitsData'

    return newData
    


    [docs]def normalize_modulated_FitsData(fitsData):
    """Normalize all HDUs for each modulator state so that the mean is the same as the global mean

    In other words, each modulator state is normalized to a mean of 1 and then all modulator states are scaled by the
    original, total mean.

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Data to be normalized. No problem is each modulator state has a different number of HDUs
    """
    nummod = fitsData.headers[0][S122['NumberOfModulatorStates']]

    # First, we'll compute the "global" mean
    global_sum = 0.
    for m in range(nummod):
        mod_sum = 0.
        mod_idx = fitsData.find_data_keyword_idx(S122['CurrentState'], m)
        for j in mod_idx:
            mod_sum += np.nanmean(fitsData.data[j])

        global_sum += mod_sum / len(mod_idx)

    global_mean = global_sum / nummod
    print('{}: normalizing all modulator states to mean of {:.3f}'.format(tag(), global_mean))

    # Now normalize each modulator state
    for m in range(nummod):
        mod_idx = fitsData.find_data_keyword_idx(S122['CurrentState'], m)
        for j in mod_idx:
            norm = np.nanmean(fitsData.data[j])
            if norm == 0:
                print('{}: HDU {} has a mean of 0 (weird). Skipping this HDU'.format(tag(), j))
                continue
            fitsData.data[j] = fitsData.data[j] / norm * global_mean
            if fitsData.unc is not None:
                fitsData.unc[j] = fitsData.unc[j] / norm * global_mean
    



    [docs]def dark_correction(fitsData, dark_calibration):
    """Subtract a Dark from input data. This function does not care about modulation state.

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Data to be corrected

    dark_calibration : `Data.FitsData`
        Dark to be subtracted

    Returns
    -------
    None, correction is done in place
    """
    print('{}: Performing dark correction on {} with {}'.format(tag(), fitsData, dark_calibration))
    dark_data = dark_calibration.data[0]
    try:
        dark_unc = dark_calibration.unc[0]
    except:
        raise ValueError('Error loading dark uncertainty data')

    for i in range(len(fitsData.data)):
        print('{}: subtracting from data HDU {}'.format(tag(), i))
        fitsData.data[i] = fitsData.data[i] - dark_data
        if fitsData.unc is not None:
            fitsData.unc[i] = np.sqrt(fitsData.unc[i]**2 + dark_unc**2)

    # Pretty hacky way to do this, and probably not very memory efficient, but I don't expect this to happen very
    # often anyway
    if fitsData.unc is None:
        fitsData.set_unc(np.dstack([dark_unc] * len(fitsData.data)))
    


    [docs]def unmodulated_gain_correction(fitsData, gain_calibration):
    """Divide ALL data arrays by provided gain.

    Only the first Gain data array is used and this function does not care about modulator state.

    If the data and gain were taken with different Cryo attenuation filters a scalar correction will be applied.

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Data to be corrected

    gain_calibration : `Data.FitsData`
        Gain to use for correction. If multiple data HDUs are present only the first will be used.

    Returns
    -------
    None, correction is done in place
    """
    print('{}: Performing gain correction on {} with {}'.format(tag(), fitsData, gain_calibration))

    if gain_calibration.unc is None:
        print('{}: WARNING! Gain object does not have uncertainties. Setting to 0, but this is weird!'.format(tag()))
        gain_calibration.set_unc(np.zeros(gain_calibration.data[0].shape + (len(gain_calibration.data), )))

    # TODO: Raise error/warning if gain_calibration is modulated
    gain_data = gain_calibration.data[0]
    gain_unc = gain_calibration.unc[0]

    data_ND_name = fitsData.headers[0][S122['AttenuationFilterID']]
    gain_ND_name = gain_calibration.headers[0][S122['AttenuationFilterID']]
    if gain_ND_name != data_ND_name:
        print('{}: Data ({}) and Gain ({}) ND filter IDs are different'.format(tag(), data_ND_name, gain_ND_name))
        try:
            data_curve = ND_CURVES[data_ND_name]
        except KeyError:
            raise KeyError('Cannot find {} (from data) in ND filter curve database'.format(data_ND_name))
        try:
            gain_curve = ND_CURVES[gain_ND_name]
        except KeyError:
            raise KeyError('Cannot find {} (from gain) in ND filter curve database'.format(gain_ND_name))

        data_attenuation = np.interp(fitsData.headers[0][S122['Wavelength']], data_curve[0], data_curve[1])
        gain_attenuation = np.interp(gain_calibration.headers[0][S122['Wavelength']], gain_curve[0], gain_curve[1])

        ND_correction = gain_attenuation / data_attenuation
        print('{:}:  ND filter correction = {:.3f}'.format(tag(), ND_correction))
    else:
        ND_correction = 1.

    for i in range(len(fitsData.data)):
        print('{}: dividing gain from data HDU {}'.format(tag(), i))
        if fitsData.unc is not None:
            # We propagate uncertainties first because the data are modified in place
            fitsData.unc[i] = ND_correction * np.sqrt((fitsData.unc[i] / gain_data)**2 +
                                                      (gain_unc * fitsData.data[i] / gain_data**2)**2)
        fitsData.data[i] = ND_correction * fitsData.data[i] / gain_data

    if fitsData.unc is None:
        fitsData.set_unc(np.dstack([gain_unc] * len(fitsData.data)))
    


    [docs]def modulated_gain_correction(fitsData, gain_calibration):
    """Divide each modulator state's data by Gain from the same modulator state.

    Every modulator state present in the input data must have a corresponding array in the Gain data.

    If the data and gain were taken with different Cryo attenuation filters a scalar correction will be applied.

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Data to correct

    gain_calibration : `Data.FitsData`
        Gain to use for correction. Should have multiple data HDUs corresponding to multiple modulator states

    Returns
    -------
    None, correction is done in place
    """
    print('{}: Performing gain correction on {} with {}'.format(tag(), fitsData, gain_calibration))
    nummod = fitsData.headers[0][S122['NumberOfModulatorStates']]

    try:
        if fitsData.headers[0]['ISSTOKES']:
            raise ValueError('You are trying to correct unmodulated data. Use generic.unmodulated_gain_correction instead')
    except KeyError:
        pass
    try:
        if gain_calibration.headers[0]['ISSTOKES']:
            raise ValueError('You are trying to correct modulated data with a unmodulated Calibration. '
                             'Use generic.unmodulated_gain_correction instead')
    except KeyError:
        pass

    if fitsData.unc is None:
        fresh_unc = np.zeros(fitsData.data[0].shape + (len(fitsData.data), ))

    if gain_calibration.unc is None:
        print('{}: WARNING! Gain object does not have uncertainties. Setting to 0, but this is weird!'.format(tag()))
        gain_calibration.set_unc(np.zeros(gain_calibration.data[0].shape + (len(gain_calibration.data), )))

    data_ND_name = fitsData.headers[0][S122['AttenuationFilterID']]
    gain_ND_name = gain_calibration.headers[0][S122['AttenuationFilterID']]
    if gain_ND_name != data_ND_name:
        print('{}: Data ({}) and Gain ({}) ND filter IDs are different'.format(tag(), data_ND_name, gain_ND_name))
        try:
            data_curve = ND_CURVES[data_ND_name]
        except KeyError:
            raise KeyError('Cannot find {} (from data) in ND filter curve database'.format(data_ND_name))
        try:
            gain_curve = ND_CURVES[gain_ND_name]
        except KeyError:
            raise KeyError('Cannot find {} (from gain) in ND filter curve database'.format(gain_ND_name))

        data_attenuation = np.interp(fitsData.headers[0][S122['Wavelength']], data_curve[0], data_curve[1])
        gain_attenuation = np.interp(gain_calibration.headers[0][S122['Wavelength']], gain_curve[0], gain_curve[1])

        ND_correction = gain_attenuation / data_attenuation
        print('{:}:  ND filter correction = {:.3f}'.format(tag(), ND_correction))
    else:
        ND_correction = 1.

    for i in range(nummod):
        print('{}: modulator state {}'.format(tag(), i))
        didx = fitsData.find_data_keyword_idx(S122['CurrentState'], i)
        gidx = gain_calibration.find_data_keyword_idx(S122['CurrentState'], i)
        if len(gidx) == 0:
            raise ValueError('Could not find gain data for modulator state {}'.format(i))
        if len(gidx) > 1:
            print('{}: WARNING! Gain object has more than one HDU for modulator state {}. This is weird'.format(tag(), i))

        gain_data = gain_calibration.data[gidx[0]]
        gain_unc = gain_calibration.unc[gidx[0]]
        for j in didx:
            print('{}: dividing gain from data HDU {}'.format(tag(), j))
            if fitsData.unc is not None:
                # We propagate uncertainties first because the data are modified in place
                fitsData.unc[j] = ND_correction * np.sqrt((fitsData.unc[j] / gain_data)**2 +
                                                          (gain_unc * fitsData.data[j] / gain_data**2)**2)
            else:
                fresh_unc[:, :, j] = gain_unc
            fitsData.data[j] =  ND_correction * fitsData.data[j] / gain_data

    if fitsData.unc is None:
        fitsData.set_unc(fresh_unc)
    


    [docs]def sort_hdus_by_modulator(fitsData):
    """Make sure that the HDUList is in modulator order.

    We also check that there are the correct number of modulator states and no repeats of a single state.
    """

    sort_idx = [0]  # Because the first HDU (Primary) will *always* be first.
    mod_list = [h.header[S122['CurrentState']] for h in fitsData.hdulist[1:]]
    if len(np.unique(mod_list)) != len(mod_list):
        raise ValueError('There are more than 1 HDU for some modulator states')

    if len(mod_list) != fitsData.headers[0][S122['NumberOfModulatorStates']]:
        raise ValueError('Expecting {} modulator states, but only found {}'.format(fitsData.headers[0][S122['NumberOfModulatorStates']],
                                                                                   len(mod_list)))

    sort_idx += list(np.argsort(mod_list) + 1)  # +1 because of the default first value
    fitsData.hdulist = pyfits.HDUList([fitsData.hdulist[i] for i in sort_idx])

    try:
        fitsData.file_list[:] = [fitsData.file_list[i - 1] for i in sort_idx[1:]]
    except AttributeError:
        pass
