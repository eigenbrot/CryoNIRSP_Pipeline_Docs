         **align_params):

    if linearize:
        lin_dir = os.path.join(pinhole_dir, 'linearized')
        LinearityCorrection.main(pinhole_dir, lin_dir, **CONSTANTS['LinearityKwargs'])
        pinhole_dir = lin_dir

    # TODO: CHANGE THIS ONCE WE GET REAL ALIGN3 DATA
    pinholeData = Data.SPAlign1Data(pinhole_dir)
    if len(pinholeData.data) > 1:
        # This might change at some point, but probably not
        pinholeData = generic.average_FitsData(pinholeData)

    print('{}: Computing beam angles'.format(tag()))
    angles = find_beam_disp_angles(pinholeData, extra_clean=angle_clean, debug=debug)

    if linearize:
        CONSTANTS['LinearityKwargs']['threshold_file'] = None
        lin_dir = os.path.join(align_dir, 'linearized')
        LinearityCorrection.main(align_dir, lin_dir, **CONSTANTS['LinearityKwargs'])
        align_dir = lin_dir

    # TODO: CHANGE THIS ONCE WE GET REAL ALIGN3 DATA
    alignData = Data.Gain4Data(align_dir)
    if len(alignData.data) > 1:
        # This will likely change in the future once real ALIGN3 data are provided/used
        alignData = generic.average_FitsData(alignData)

    validate_align3_task_data(pinholeData, alignData)

    print('{}: Computing beam mapping transform'.format(tag()))
    fit_output = find_beam_align_transform(alignData, extra_clean=align_clean, debug=debug, **align_params)

    validate_align3_calibration(fit_output)

    print('{}: Generating Align3 Calibration'.format(tag()))
    alignCal = generate_align3_calibration(angles, fit_output, output)

    return alignCal

    [docs]def validate_align3_task_data(pinholeData, alignData):
    """Right now this just ensures that all the IP's are the same.

    Checks for the correct task and subtask are already handled by the _TaskData derivatives used to find the data.
    """

    IP0 = pinholeData.headers[1][S122['InstrumentProgramID']]
    for h in pinholeData.data_hdus:
        if h.header[S122['InstrumentProgramID']] != IP0:
            raise ValueError('Not all pinhole frames have the same IP ID')

    IP0 = alignData.headers[1][S122['InstrumentProgramID']]
    for h in alignData.data_hdus:
        if h.header[S122['InstrumentProgramID']] != IP0:
            raise ValueError('Not all align3 frames have the same IP ID')

    return
    


    [docs]def validate_align3_calibration(fit_output):
    """Check that the affine transform isn't weird.
    """
    parvals = fit_output.params.valuesdict()
    for scale in ['scale_x', 'scale_y']:
        if parvals[scale] < CONSTANTS['Align3Calibration']['min_scale'] or \
           parvals[scale] > CONSTANTS['Align3Calibration']['max_scale']:
            raise ValueError('X scale ({}) is out of range'.format(parvals[scale]))
    



    [docs]def generate_align3_calibration(angles, fit_output, outname):
    """Collect beam angles and mapping into a single object and write to disk

    For now the object is just a dictionary and it is saved to disk in an ASDF file

    Parameters
    ----------
    angles : `numpy.ndarray`
        Vector containing angles dispersion angles for beam 1 and beam2

    fit_output : `lmfit.MinimizerResult`
        Object containing best-fit affine transform parameters and additional fitting statistics/information

    outname : str
        Location to save align3 parameters

    Returns
    -------
    dict:
        The Align3 Calibration

    """
    parvals = fit_output.params.valuesdict()
    transform = skit.AffineTransform(scale=[parvals['scale_x'], parvals['scale_y']],
                                     rotation=parvals['rotation'], shear=parvals['shear'],
                                     translation=[parvals['translation_x'], parvals['translation_y']])
    tree = {'angles': {
                       'beam1': angles[0],
                       'beam2': angles[1]
                       },
            'transform': {
                          'type': 'affine',  # Will this ever be different?
                          'matrix': transform.params,
                          'scale_x': transform.scale[0],
                          'scale_y': transform.scale[1],
                          'translation_x': transform.translation[0],
                          'translation_y': transform.translation[1],
                          'rotation': transform.rotation,
                          'shear': transform.shear
                          }
           }

    adf = asdf.AsdfFile(tree)
    adf.write_to(outname)

    return tree
    


    [docs]def find_beam_disp_angles(fitsData, extra_clean=False, debug=False):
    """Measure the angle between the dispersion and detector axes for each beam

    This is done with pinhole image data and uses a pretty straight-forward Hough Transform to identify the strongest
    angle.

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Object containing a single pinhole frame for BOTH BEAMS.

    extra_clean : bool
        If True then an additional, expensive cleaning step will be done before performing the Hough Transform. This is
        very good at getting rid of salty bad-pixel data, but in practice this type of noise probably doesn't affect
        the Hough Transform too much.

    debug : bool
        If True then various debug images and plots will be produced. The code will pause between beams so that these
        results can be examined.

    Returns
    -------
    `numpy.ndarray`
        Vector of two numbers; the angle (in degrees) between the dispersion and detector axes for each beam,
        respectively.
    """
    data = fitsData.data[0]
    beam1, beam2 = split_beams(data)

    angles = []
    for i, beam in enumerate([beam1, beam2]):
        print('{}: Working on beam {}'.format(tag(), i))
        binary = make_binary(beam, extra_clean=extra_clean, debug=debug)
        H, theta, rho = do_hough(binary, debug=debug)
        peak_theta, _ = find_px_angles(H, theta, debug=debug)
        if debug:
            input('Debug images made. Press enter to continue')
        angles.append(peak_theta)

    return -1 * np.rad2deg(np.array(angles))
    



    [docs]def find_beam_align_transform(fitsData, extra_clean=True, method='lbfgsb', perturb_params=False,
                              vary_scale=True, vary_shear=False, vary_rotation=False, debug=False):
    """
    Compute the Affine mapping of Beam 2 onto Beam 1

    The (x, y) translation is always fit, and the scaling, rotation, and shear can be fit if desired.

    An initial guess is made using the points in CryoNIRSP_Pipeline.data.ALIGN3_NUMBERS. The default options
    should be the best choice in most/all cases.

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Object containing the full-frame detector data for a single frame. This should either be a Solar Gain or an
        Align3 Program Task.

    extra_clean : bool
        If True then an additional, expensive cleaning step will be done to reduce the amount of pepper-like noise in
        the data.

    method : str
        The fitting method used to minimize affine parameters. Valid values are:
        - `'differential_evolution'`: differential evolution (default)
        - `'leastsq'`: Levenberg-Marquardt
        - `'least_squares'`: Least-Squares minimization, using Trust Region Reflective method by default
        - `'brute'`: brute force method
        - '`nelder`': Nelder-Mead
        - `'lbfgsb'`: L-BFGS-B
        - `'powell'`: Powell
        - `'cg'`: Conjugate-Gradient
        - `'newton'`: Newton-Congugate-Gradient
        - `'cobyla'`: Cobyla
        - `'tnc'`: Truncate Newton
        - `'trust-ncg'`: Trust Newton-Congugate-Gradient
        - `'dogleg'`: Dogleg
        - `'slsqp'`: Sequential Linear Squares Programming

        For more details on the fitting methods please refer to the
        `SciPy docs <http://docs.scipy.org/doc/scipy/reference/optimize.html>`__.

    perturb_params : bool
        If True then the initial Affine Transform is perturbed by up to 10%. This can improve accuracy for some fitting
        algorithms if the initial guess is really good.

    vary_scale : bool
        If True then allow the x and y scale factors to vary during the fit

    vary_shear : bool
        If True then allow the shear factor to vary during the fit

    vary_rotation : bool
        If True then allow the rotation to vary during the fit

    debug : bool
        If True then create and save FITS files of both beams at all steps in the fitting process. Their names and
        additional information will be printed.

    Returns
    -------
    `lmfit.MinimizerResult`
        Object containing best-fit affine parameters and additional fitting statistics/information
    """
    data = fitsData.data[0]
    beam1, beam2 = split_beams(data)

    beam1 = prep_data(beam1)
    beam2 = prep_data(beam2)

    if extra_clean:
        beam1 = clean_pepper(beam1)
        beam2 = clean_pepper(beam2)

    print('{}: Computing initial transform'.format(tag()))
    initial_transform = skit.estimate_transform('affine',
                                                np.array(ALIGN3_NUMBERS['Beam_2_match_points']),
                                                np.array(ALIGN3_NUMBERS['Beam_1_match_points']))

    print('{}: Generating parameters'.format(tag()))
    params = generate_parameters(initial_transform, vary_scale=vary_scale, vary_shear=vary_shear,
                                 vary_rotation=vary_rotation)

    if perturb_params:
        print('{}: Perturbing initial guesses'.format(tag()))
        perturb_parameters(params)

    if debug:
        pyfits.PrimaryHDU(beam1).writeto('fit_input_beam1.fits', overwrite=True)
        pyfits.PrimaryHDU(beam2).writeto('fit_input_beam2.fits', overwrite=True)
        print("{}: Fit input beams saved as 'fit_input_beam[12].fits'".format(tag()))

        beam2_trans = transform_beam(params, beam2, beam1.shape)
        pyfits.PrimaryHDU(beam2_trans).writeto('fit_initial_guess.fits', overwrite=True)
        print("{}: Fit initial guess saved as 'fit_initial_guess.fits'".format(tag()))
        print('{}: Initial transform beam similarity: {:.1f}%'.format(tag(), 100 * (1 - compare_beams(params, beam1, beam2))))

    print('{}: Starting minimizer with free parameters {}'.format(tag(), [p for p in params if params[p].vary]))
    mini = lmfit.Minimizer(compare_beams, params, fcn_args=(beam1, beam2))
    t1 = time.time()
    fit_out = mini.minimize(method=method, params=params)
    print('{}: Minimization complete in {:.1f} s'.format(tag(), time.time() - t1))

    if debug:
        beam2_trans = transform_beam(fit_out.params, beam2, beam1.shape)
        pyfits.PrimaryHDU(beam2_trans).writeto('fit_beam2_output.fits', overwrite=True)
        print("{}: Best-fit beam2 transform saved as 'fit_beam2_output.fits'".format(tag()))
        print('{}: Final transform beam similarity: {:.1f}%'.format(tag(), 100 * (1 - compare_beams(fit_out.params, beam1, beam2))))

    return fit_out
    



    [docs]def generate_parameters(transform, vary_scale=False, vary_shear=False, vary_rotation=False):
    """Create fit Parameters based on an initial-guess transform

    Parameters
    ----------
    transform : `skimage.transform._geometric.AffineTransform`
        Initial-guess transform

    vary_scale : bool
        If True the scale will be varied during the fit

    vary_shear : bool
        If True the shear will be varied during the fit

    vary_rotation : bool
        If True the rotation will be varied during the fit

    Returns
    -------
    `lmfit.parameter.Parameters`
        Set of Parameters suitable for fitting with lmfit
    """
    params = lmfit.Parameters()

    params.add('scale_x',
               value=transform.scale[0],
               min=ALIGN3_NUMBERS['Scale_x']['min'],
               max=ALIGN3_NUMBERS['Scale_x']['max'],
               vary=vary_scale)
    params.add('scale_y',
               value=transform.scale[1],
               min=ALIGN3_NUMBERS['Scale_y']['min'],
               max=ALIGN3_NUMBERS['Scale_y']['max'],
               vary=vary_scale)

    params.add('translation_x',
               value=transform.translation[0],
               min=ALIGN3_NUMBERS['Translation_x']['min'],
               max=ALIGN3_NUMBERS['Translation_x']['max'])
    params.add('translation_y',
               value=transform.translation[1],
               min=ALIGN3_NUMBERS['Translation_y']['min'],
               max=ALIGN3_NUMBERS['Translation_y']['max'])

    params.add('rotation',
               value=transform.rotation,
               min=np.deg2rad(ALIGN3_NUMBERS['Rotation']['min']),
               max=np.deg2rad(ALIGN3_NUMBERS['Rotation']['max']),
               vary=vary_rotation)

    params.add('shear',
               value=transform.shear,
               min=ALIGN3_NUMBERS['Shear']['min'],
               max=ALIGN3_NUMBERS['Shear']['max'],
               vary=vary_shear)

    return params
    


    [docs]def perturb_parameters(params):
    """Nudge initial-guess parameters

    Some fitting algorithms may be more accurate if the starting point is not right on the best-fit. This function
    nudges the initial guess in case it's too good. What?

    Parameters
    ----------
    params : `lmfit.parameter.Parameters`
        The Parameters to nudge. Only those that vary will be updated.
    """
    rng = np.random.default_rng()
    for p in params:
        if params[p].vary:
            width = params[p].max - params[p].min
            pert = width * (0.2 * rng.uniform() - 0.1)
            print('{}: Perturbing {} by {:.3f}'.format(tag(), p, pert))
            params[p].value = params[p].value + pert
    


    [docs]def compare_beams(params, beam1, beam2):
    """Compute the goodness-of-fit parameter for 2 beams.

    We use the structural similarity (SSIM) to measure how similar the two images are.

    To avoid mis-fits caused by different beam sizes/FOV's we only compare the portions of the beams that overlap each
    other after beam 2 is transformed.

    Parameters
    ----------
    params : `lmfit.parameter.Parameters`
        The fitting parameters

    beam1 : `numpy.ndarray`
        Beam 1, the reference image

    beam2 : `numpy.ndarray`
        Beam 2 the image to be transformed

    Returns
    -------
    float
        The SSIM between beam 1 and transformed beam 2
    """
    beam2_trans = transform_beam(params, beam2, beam1.shape)

    try:
        ssim = 1 - skme.structural_similarity(beam1[~np.isnan(beam2_trans)],
                                              beam2_trans[~np.isnan(beam2_trans)])
    except ValueError:
        # This happens if the transform's cval makes all of beam2_trans NaN's. Then the above slicing will cause
        #  the calculation to fail on an empty slice. In any case, these beams ain't similar at all!
        return 1.

    return ssim
    



    [docs]def transform_beam(params, beam, shape):
    """Actually transform beam 2 based on the fit parameters
    """
    parvals = params.valuesdict()
    transform = skit.AffineTransform(scale=[parvals['scale_x'], parvals['scale_y']],
                                     rotation=parvals['rotation'], shear=parvals['shear'],
                                     translation=[parvals['translation_x'], parvals['translation_y']])
    beam_trans = skit.warp(beam, transform.inverse, preserve_range=True, output_shape=shape, cval=np.nan)

    return beam_trans
    



    [docs]def prep_data(beam):
    """Prepare the data for fitting by removing NaN's, normalizing, and casting as float64.
    """
    beam /= np.nanmean(beam)
    beam[np.isnan(beam)] = np.nanmedian(beam)

    return beam.astype(np.float64)
    



    [docs]def split_beams(data):
    """Perform a simple and rough separation of the two polarimetric beams

    This is just a simple array slice right now. It might get more complicated, but it might not!

    Parameters
    ----------
    data : `numpy.ndarray`
        Full frame data to split

    Returns
    -------
    `numpy.ndarray`
        Data array for beam 1

    `numpy.ndarray`
        Data array for beam 2
    """
    pivot = ALIGN3_NUMBERS['Beam_border']
    print('{}: Splitting beams along column {}'.format(tag(), pivot))

    beam1 = data[:, :pivot]
    beam2 = data[:, pivot:]

    return beam1, beam2
    



    [docs]def make_binary(data, numotsu=2, extra_clean=False, debug=False):
    # (Mostly) copied from ViSP_Pipeline.TargetCalibration
    """Generate a binary image from an input float or int array.

    A threshold value is chosen via Otsu's method and all values below the threshold are set to 0 while everything
    else is set to 1.

    Parameters
    ----------
    data : numpy.ndarray
        Data to convert to binary

    numotsu : int
        The number of times to perform thresholding using Otsu's method. Numbers larger than 1 are useful if the
        pixel distribution of the data has > 2 modes.

    debug : bool
        If True then the resulting binary image will be saved as "binary.fits" in the current directory.

    Returns
    -------
    numpy.ndarray
        An array of 1's and 0's corresponding to the thresholded input data.

    """
    threshold = -np.inf
    for i in range(numotsu):
        print('{}: Threshold on iteration {} = {}'.format(tag(), i, threshold))
        threshold = skif.threshold_otsu(data[data > threshold])

    binary = np.array(data > threshold, dtype=np.int)

    if extra_clean:
        binary = clean_salt(binary)

    if debug:
        pyfits.ImageHDU(binary).writeto('binary.fits', overwrite=True)

    return binary

    [docs]def do_hough(binary, numtheta=1500, debug=False):
    # Copied from ViSP_Pipeline.TargetCalibration
    """Perform a Hough line transform on in the input data.

    Resolution in the angle dimension can be increased via numtheta.

    Parameters
    ----------
    binary : numpy.ndarray
        An integer array of 1's and 0's

    numtheta : int
        The number of samples in the theta dimension of the Hough transform.

    debug : bool
        If True a FITS file of the resulting Hough space will be written as "hough.fits"

    Returns
    -------
    numpy.ndarray
        The 2D Hough space (i.e., accumulator) of the input image

    numpy.ndarray
        A 1D array of the angle values corresponding to the Hough accumulator. Angles will range from -pi/4 to 3/4 pi
        and the length will be equal to numtheta.

    numpy.ndarray
        A 1D array of the distance values corresponding to the Hough accumulator. This array is entirely determined
        by skimage.transform.hough_line

    """
    hough = skit.hough_line(binary, theta=np.linspace(-np.pi / 4, 0.75 * np.pi, numtheta))
    H = hough[0]
    theta = hough[1]
    rho = hough[2]

    if debug:
        pyfits.PrimaryHDU(H).writeto('hough.fits', overwrite=True)

    return H, theta, rho
    



    [docs]def find_px_angles(H, theta, theta_min=None, theta_max=None, debug=False):
    # Copied from ViSP_Pipeline.TargetCalibration
    """Find the most significant angles in a Hough transform.

    Peaks in the Hough transform a first found with a simple max filter and then refined via interpolation of the
    surrounding peak.

    Note, peak angles are not allowed to be within pi/10 of each other.

    Parameters
    ----------
    H : numpy.ndarray
        A 2D array representing a Hough accumulator

    theta : numpy.ndarray
        A 1D array of the angle values corresponding to H

    debug : bool
        If True a plot showing the peak angles and interpolated values will be saved as "peaks.png" in the current
        directory.

    Returns
    -------
    numpy.ndarray
        The most significant angels found

    numpy.ndarray
        The index values corresponding to the location in theta of the initial (non-interpolated) guesses for most
        significant angles.

    """
    if theta_min is None:
        theta_min = np.min(theta)
    if theta_max is None:
        theta_max = np.max(theta)

    range_idx = np.where((theta >= theta_min) & (theta <= theta_max))[0]

    fit_theta = theta[range_idx]
    rss = np.sqrt(np.sum(H[:, range_idx]**2, axis=0))

    # This min distance does not consider any angles within pi/10 of each other
    idx = pku.indexes(rss, min_dist=theta.size / 10)

    # We subtract by pi/2 because Hough measures angle from spatial axis and we want angle from dispersion axis
    peak_theta = pku.interpolate(fit_theta, rss, ind=idx)[0] - np.pi / 2.

    print('{}: found peak in theta RSS at {}'.format(tag(), idx))
    print('{}: most prominent angle (deg): {}'.format(tag(), np.rad2deg(peak_theta)))

    if debug:
        ax = plt.figure().add_subplot(111)
        ax.plot(fit_theta, rss, '.')
        ax.axvline(x=peak_theta)
        ax.figure.savefig('peaks.png')

    return peak_theta, idx
